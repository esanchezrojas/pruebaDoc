import { Injectable } from '@angular/core';
import { HttpClient, HttpErrorResponse } from '@angular/common/http';
import { throwError } from 'rxjs';
import { ResponseModel } from '@core/models/response-model/ResponseModel';
import { environment } from '@environments/environment';
import { catchError } from 'rxjs/operators';
import { Proveedor } from '../../../proveedores/interfaces/proveedor';
import { TreeNode } from 'primeng/api';

interface ProductoP {
  producto_id: number;
}

@Injectable({
  providedIn: 'root'
})
export class ProveedoresService {

  static handleError(error: HttpErrorResponse): any {
    console.log(error);
    return throwError('Ha ocurrido un error..');
  }

  constructor(private http: HttpClient) { }

  getProveedores(): any {
    const dataObj: object = {
      metodo: 'getProveedores',
    };
    return this.http.post<any>(`${environment.url_api}/proveedoresCtrl.php`, dataObj).toPromise().then(res => res.dataObjet as TreeNode[]);
  }

  getProducto(proveedorId): any {
    const dataObj: object = {
      metodo: 'getProducto',
      proveedor: proveedorId
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/proveedoresCtrl.php`, dataObj).pipe(
      catchError(ProveedoresService.handleError));
  }

  saveProducto(proveedorId, producto: ProductoP): any{
    const payLoad: object = {
      metodo: 'saveProduct',
      proveedor: proveedorId,
      data: producto
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/proveedoresCtrl.php`, payLoad).pipe(
      catchError(ProveedoresService.handleError));

  }

  saveContacto(contacto: object): any{
    const payLoad: object = {
      metodo: 'saveContacto',
      data: contacto
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/proveedoresCtrl.php`, payLoad).pipe(
      catchError(ProveedoresService.handleError));

  }

  saveProveedor(proveedor: Proveedor): any{
    const payLoad: object = {
      metodo: 'saveProveedor',
      data: proveedor
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/proveedoresCtrl.php`, payLoad).pipe(
      catchError(ProveedoresService.handleError));

  }

}
