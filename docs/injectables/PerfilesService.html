import { Injectable } from '@angular/core';
import {ResponseModel} from "@core/models/response-model/ResponseModel";
import {environment} from "@environments/environment";
import {catchError} from "rxjs/operators";
import {HttpClient, HttpErrorResponse} from "@angular/common/http";
import {throwError} from "rxjs";
import {TreeNode} from "primeng/api";

interface DataNavegacion {
  proceso_id: number;
  aplicacion_id: number;
  menu_id: number;
  id: number;
  descripcion: string;
  estado: string;
  tipo: string;
  ruta: string;
}

@Injectable({
  providedIn: 'root'
})
export class PerfilesService {

  static handleError(error: HttpErrorResponse): any {
    console.log(error);
    return throwError('Ha ocurrido un error..');
  }

  constructor(private http: HttpClient) { }

  getProcesos(Filtro: any): any {
    const dataObj: object = {
      metodo: 'listaProceso',
      proceso_id: Filtro.proceso
    };
    return this.http.post<any>(`${environment.url_api}/perfilCtrl.php`, dataObj).toPromise().then(res => <TreeNode[]>res.dataObjet);
  }

  getPerfiles(Filtro: any): any {
    const dataObj: object = {
      metodo: 'getPerfiles',
      perfil_aplicacion: Filtro.proceso
    };
    return this.http.post<any>(`${environment.url_api}/perfilCtrl.php`, dataObj).toPromise().then(res => <TreeNode[]>res.dataObjet);
  }

  getProcesoApp(Filtro: any): any {
    const data: object = {
      metodo: 'getPerfiles',
      perfil_aplicacion: Filtro.proceso
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  getSelectProcesos(): any {
    const data: object = {
      metodo: 'getSelectProcesos'
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  getSelectMenuApp(app: number): any {
    const data: object = {
      metodo: 'getSelectMenuApp',
      app: app
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  getCargosPick(perfilApp: number): any {
    const data: object = {
      metodo: 'getCargosPick',
      app: perfilApp
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  addDataNavegacion(dataPerfil: DataNavegacion): any {
    const data: object = {
      metodo: 'addDataNavegacion',
      data: dataPerfil
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  addDataAppPerfil(dataPerfil: any): any {
    const data: object = {
      metodo: 'addDataAppPerfil',
      data: dataPerfil
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  editDataPerfilApp(dataPerfil: any): any {
    const data: object = {
      metodo: 'editPerfilApp',
      data: dataPerfil
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  addPerfilMenu(dataPerfil: any): any {
    const data: object = {
      metodo: 'addPerfilMenu',
      data: dataPerfil
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  editDataNavegacion(dataPerfil: DataNavegacion): any {
    const data: object = {
      metodo: 'editDataNavegacion',
      data: dataPerfil
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  editEstadoNavegacion(estado: string, dataPerfil: DataNavegacion): any {
    const data: object = {
      metodo: 'editEstadoNavegacion',
      data: dataPerfil,
      estado: estado
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  editEstadoPerfilApp(estado: string, dataPerfil: any): any {
    const data: object = {
      metodo: 'editEstadoPerfilApp',
      data: dataPerfil,
      estado: estado
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  editTargetCargo(items: any, tipo: string): any {
    const data: object = {
      metodo: 'editTargetCargo',
      items: items,
      tipo: tipo
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }

  editPermisoMenu(valorN: string, tipoE: string, dataPerfil: any): any {
    const data: object = {
      metodo: 'editPermisoMenu',
      data: dataPerfil,
      valorN: valorN,
      tipoE: tipoE
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/perfilCtrl.php`, data).pipe(
      catchError(PerfilesService.handleError)
    );
  }
}
