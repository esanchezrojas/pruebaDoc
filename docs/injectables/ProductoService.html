import { Injectable } from '@angular/core';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {environment} from '@environments/environment';
import {catchError} from 'rxjs/operators';
import {HttpClient, HttpErrorResponse} from '@angular/common/http';
import {throwError} from 'rxjs';
interface Producto {
  descripcion: string;
  clasificador1: number;
  clasificador2: number;
  clasificador3: number;
  marca: number;
  modelo: string;
}
@Injectable({
  providedIn: 'root'
})
export class ProductoService {
  static handleError(error: HttpErrorResponse): any {
    console.log(error);
    return throwError('Ha ocurrido un error..');
  }
  constructor(private http: HttpClient) {
  }
  getClasificadorByType(type: string): any {
    const dataObj: object = {
      metodo: 'getClasificadorByType',
      tipo: type
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, dataObj).pipe(
      catchError(ProductoService.handleError));
  }
  getMarca(): any {
    const dataObj: object = {
      metodo: 'getMarca'
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, dataObj).pipe(
      catchError(ProductoService.handleError));
  }
  saveProducto(producto: Producto): any{
    const payLoad: object = {
      metodo: 'saveProduct',
      data: producto
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, payLoad).pipe(
      catchError(ProductoService.handleError));

  }
  getSede(): any {
    const dataObj: object = {
      metodo: 'getSede'
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, dataObj).pipe(
      catchError(ProductoService.handleError));
  }
  getProducto(): any {
    const dataObj: object = {
      metodo: 'getProducto'
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, dataObj).pipe(
      catchError(ProductoService.handleError));
  }
  getHDV(id): any {
    const dataObj: object = {
      metodo: 'getHDV',
      hdvid: id
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, dataObj).pipe(
      catchError(ProductoService.handleError));
  }
  saveInvent(producto: Producto): any{
    const payLoad: object = {
      metodo: 'saveInvent',
      data: producto
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, payLoad).pipe(
      catchError(ProductoService.handleError));

  }
  editProducto(producto: Producto): any{
    const dataObj: object = {
      metodo: 'editProduct',
      data: producto
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, dataObj).pipe(
      catchError(ProductoService.handleError));
  }
  listaProducto(obj: any): any {
    const dataObj: object = {
      metodo: 'getProducts',
      limit: obj.limit,
      page: obj.page,
      sort: obj.sort,
      filter: obj.filter,
      order: obj.order
    };

    return this.http.post<ResponseModel>(`${environment.url_api}/productoCtrl.php`, dataObj).pipe(
      catchError(ProductoService.handleError));
  }
}
