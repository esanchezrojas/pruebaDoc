import {ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {ConfirmationService, MessageService} from 'primeng';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {EmpleadosService} from '@core/services/empleados/empleados.service';
import {Router} from '@angular/router';
import {FormBuilder, Validators} from '@angular/forms';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {Observable} from 'rxjs';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';

interface Mascota {
  empleadomascota_id: number;
  empleado_id: number;
  nombre_mascota: string;
  fecha_nacim: string;
  clase_id: number;
  raza: string;
  estado: string;
}

@Component({
  selector: 'app-empleado-mascotas',
  templateUrl: './empleado-mascotas.component.html',
  styleUrls: ['./empleado-mascotas.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class EmpleadoMascotasComponent implements OnInit {

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private utilService: UtilService,
              private empleadoService: EmpleadosService,
              private cd: ChangeDetectorRef,
              private router: Router,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };
  dato$: Observable<State>;
  logged: Observable<boolean>;
  loading = true;
  tipoMascotas = [];
  edadMascotas = [];
  pageSize = 100;
  pageNumber = 1;
  record = 0;
  sort: string;
  order: any;
  MascotasLista = [];
  submittedForm = false;
  textSpan = 'Agregar mascota';
  accion = 'ADD';
  fechaNacim = new Date();

  filterListaMascota = this.fb.group({
    tipo: 'Admin'
  });

  EmpleadoMascota = this.fb.group({
    nombre_mascota: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    fecha_nacim: ['', Validators.compose( [Validators.required])],
    clase_id: ['', Validators.compose( [Validators.required])],
    raza: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    estado: ['A', Validators.compose( [Validators.required])],
    empleadomascota_id: ['']
  });

  getClasificadorByType(): void {
    this.loading = true;
    this.empleadoService.getClasificadorByType('CLMA').subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.tipoMascotas = data.dataArray;
      }
    });
    this.empleadoService.getClasificadorByType('EDAD').subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.edadMascotas = data.dataArray;
      }
    });
  }

  getData(): void {
    const dataObj: object = {
      limit: this.pageSize,
      page: this.pageNumber,
      sort: this.sort,
      filter: this.filterListaMascota.value,
      order: this.order
    };
    this.loading = true;
    this.empleadoService.getMascotasEmpleado(dataObj).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        console.log(this.MascotasLista);
        this.MascotasLista = [...data.dataArray];
        this.record = data.dataObjet.records;
        console.log(this.MascotasLista);
        console.log([...data.dataArray]);
        console.log(this.record);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  ngOnInit(): void {
    this.getClasificadorByType();
    this.getData();
  }

  saveMascota(): void {
    this.submittedForm = true;
    if (!this.EmpleadoMascota.valid) {
      return;
    }
    if (this.accion === 'ADD') {
      this.empleadoService.addMascotaEmpleado(this.EmpleadoMascota.value).subscribe(data => {
        this.saveMascotaResponse(data);
      }, error => {
        this.router.navigate(['/error']);
      });
    } else if (this.accion === 'EDIT') {
      this.empleadoService.updateMascotaEmpleado(this.EmpleadoMascota.value).subscribe(data => {
        this.saveMascotaResponse(data);
      }, error => {
        this.router.navigate(['/error']);
      });
    } else {
      this.router.navigate(['/error']);
    }
  }

  saveMascotaResponse(data: any): void {
    this.loading = false;
    this.response = data;
    if (data.state === 200) {
      this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
      this.EmpleadoMascota.reset();
      this.submittedForm = false;
      this.accion = 'ADD';
      this.textSpan = 'Agregar hijo';
      this.getData();
    } else if (data.state === 402) {
      Swal.fire({
        icon: 'info',
        title: 'Advertencia',
        timer: 1500,
        showConfirmButton: false,
        text: data.msg,
      }).then(() => {
        if (this.logged) {
          this.authService.logout().subscribe(datalog => {
            this.response = datalog;
            if (this.response.state === 200) {
              localStorage.removeItem('logged');
              this.store.dispatch(new fromRoot.CleanRootScope());
              window.location.href = '/login';
            } else if (this.response.state === 500) {
              this.router.navigate(['/error']);
            }
          }, error => {
            this.router.navigate(['/error']);
          });
        }
        this.router.navigate(['/home']);
      });
    } else if (data.state === 401) {
      Swal.fire({
        icon: 'info',
        title: 'Advertencia',
        timer: 1500,
        showConfirmButton: false,
        text: data.msg,
      }).then(() => {
        this.router.navigate(['/home']);
      });
    } else if (data.state === 500) {
      this.router.navigate(['/error']);
    } else {
      this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
    }
  }

  openDataEdit(mascota: Mascota): void {
    this.textSpan = 'Editar mascota NÂ°' + mascota.empleadomascota_id;
    this.accion = 'EDIT';
    this.fechaNacim = new Date(mascota.fecha_nacim);
    this.EmpleadoMascota.controls.raza.setValue(mascota.raza);
    this.EmpleadoMascota.controls.nombre_mascota.setValue(mascota.nombre_mascota);
    this.EmpleadoMascota.controls.clase_id.setValue(mascota.clase_id);
    this.EmpleadoMascota.controls.estado.setValue(mascota.estado);
    this.EmpleadoMascota.controls.empleadomascota_id.setValue(mascota.empleadomascota_id);
  }

}
