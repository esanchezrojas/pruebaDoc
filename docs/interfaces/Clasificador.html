  getData(): void {
    const dataObj: object = {
      limit: this.pageSize,
      page: this.pageNumber,
      sort: this.sort,
      filter: this.filterClasForm.value,
      order: this.order
    };
    this.loading = true;
    this.clasificadorService.getClasificador(dataObj).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.clasificadores = [...data.dataArray];
        this.record = data.dataObjet.records;
        this.updateRowGroupMetaData();
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  openEditClasificador(clasificador: Clasificador): void {
    if (clasificador.estado === 'Activo') {
      clasificador.estado = 'A';
    } else {
      clasificador.estado = 'I';
    }
    let tipo: { id: string; value: string; };
    tipo = this.selectTipo.find(data => data.value === clasificador.tipo);
    this.clasForm.controls.clasificador_id.setValue(clasificador.clasificador_id);
    this.clasForm.controls.descripcion.setValue(clasificador.descripcion);
    this.clasForm.controls.estado.setValue(clasificador.estado);
    this.clasForm.controls.tipo.setValue(tipo.id);
    this.editClasDialog = true;
  }

  onSort(): void {

  }

  sortData(e: LazyLoadEvent): void {
    this.sort = e.sortField;
    this.order = e.sortOrder;
    const num = 100;
    let i = 0;
    while (i < num) {
      if (e.rows * i === e.first) {
        this.pageNumber = i + 1;
      }
      i++;
    }
    this.pageSize = e.rows;
    this.getData();
  }

  filterInput(): void {
    this.getData();
  }

  updateRowGroupMetaData(): any {
    this.rowGroupMetadata = {};
    if (this.clasificadores) {
      for (let i = 0; i < this.clasificadores.length; i++) {
        const rowData = this.clasificadores[i];
        const brand = rowData.tipo;
        if (i === 0) {
          this.rowGroupMetadata[brand] = {index: 0, size: 1};
        } else {
          const previousRowData = this.clasificadores[i - 1];
          const previousRowGroup = previousRowData.tipo;
          if (brand === previousRowGroup) {
            this.rowGroupMetadata[brand].size++;
          } else {
            this.rowGroupMetadata[brand] = {index: i, size: 1};
          }
        }
      }
    }
  }
