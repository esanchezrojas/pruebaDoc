  setParam(): void {
    this.actualizaService.getParam(this.ParamUrl).subscribe(data => {
      this.response = data;
      if (this.response.state === 200) {

        this.empleadosLista = [...data.dataArray];

        this.EmpleadoEdit.controls.empleado_id.setValue(this.empleadosLista[0].empleado_id);

        this.cargaEditEmpleado(this.empleadosLista[0]);

        this.ciudad_id = data.dataObjet;


      } else if (this.response.state === 500) {
        this.router.navigate(['/error']);
      } else if (this.response.state === 402) {
        Swal.fire({
          icon: 'error',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: this.response.msg,
        }).then(() => {
          this.router.navigate(['/login']);
        });
      } else {
        Swal.fire({
          icon: 'error',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: this.response.msg,
        }).then(() => {
          this.reload();
        });
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  editaEmpleado(): void {
    this.loading = true;
    this.actualizaService.actualizaEmpleado(this.EmpleadoEdit.value).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.hideDialog();
          this.router.navigate(['/home']);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      // this.router.navigate(['/error']);
    });
  }

    this.router.routeReuseStrategy.shouldReuseRoute = () => false;
     this.router.onSameUrlNavigation = 'reload';
     this.router.navigate([`${this.router.url}`]);
  }
