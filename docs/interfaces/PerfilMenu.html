  addNewPerfilSubmenu(): void {
    this.loading = true;
    this.menuService.addPerfilSubMenu(this.menuPerfilForm.value).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.hideDialog();
        this.getData();
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });

          title: 'Advertencia',

  getData(): void {
    const dataObj: object = {
      limit: this.pageSize,
      page: this.pageNumber,
      sort: this.sort,
      filter: this.filterMenuPerfilForm.value,
      order: this.order
    };
    this.loading = true;
    this.menuService.getMenuPerfilData(dataObj).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        this.perfilMenus = [...data.dataArray];
        this.record = data.dataObjet.records;
        this.updateRowGroupMetaData();

  updateRowGroupMetaData(): any {
    this.rowGroupMetadata = {};
    if (this.perfilMenus) {
      for (let i = 0; i < this.perfilMenus.length; i++) {
        const rowData = this.perfilMenus[i];
        const brand = rowData.perfil;
        if (i === 0) {
          this.rowGroupMetadata[brand] = {index: 0, size: 1};
        } else {
          const previousRowData = this.perfilMenus[i - 1];
          const previousRowGroup = previousRowData.perfil;
          if (brand === previousRowGroup) {
            this.rowGroupMetadata[brand].size++;
          } else {
            this.rowGroupMetadata[brand] = {index: i, size: 1};
          }
        }
      }
    }
  }

  filterInput(): void {
    this.getData();
  }

  onSort(): void {
    this.updateRowGroupMetaData();
  }

  sortData(e: LazyLoadEvent): void {
    this.sort = e.sortField;
    this.order = e.sortOrder;
    const num = 100;
    let i = 0;
    while (i < num) {
      if (e.rows * i === e.first) {
        this.pageNumber = i + 1;
      }
      i++;
    }
    this.pageSize = e.rows;
    this.getData();
  }

  exportExcel(): void {
    this.utilService.exportExcel(this.perfilMenus, 'perfil-menus');
  }
