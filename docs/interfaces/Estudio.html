import {ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {ConfirmationService, MessageService} from 'primeng';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {EmpleadosService} from '@core/services/empleados/empleados.service';
import {Router} from '@angular/router';
import {FormBuilder, Validators} from '@angular/forms';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {Observable} from 'rxjs';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';

interface Estudio {
  empleadoestudio_id: number;
  empleado_id: number;
  nivelestudio_id: number;
  carrera: string;
  semestre_terminado: number;
  fecha_inicio: string;
  fecha_final: string;
  en_curso: string;
  fecha_crea: string;
  fecha_modif: string;
}

@Component({
  selector: 'app-empleado-estudios',
  templateUrl: './empleado-estudios.component.html',
  styleUrls: ['./empleado-estudios.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class EmpleadoEstudiosComponent implements OnInit {

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private utilService: UtilService,
              private empleadoService: EmpleadosService,
              private cd: ChangeDetectorRef,
              private router: Router,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };
  dato$: Observable<State>;
  logged: Observable<boolean>;
  loading = true;
  submittedForm = false;
  textSpan = 'Agregar vehículo';
  accion = 'ADD';
  pageSize = 100;
  pageNumber = 1;
  record = 0;
  sort: string;
  order: any;
  estudiosLista = [];
  tipoEstudio = [];

  fechaInicio = new Date();
  fechaFinal = new Date();

  filterListaEstudio = this.fb.group({
    tipo: 'Admin'
  });

  EmpleadoEstudio = this.fb.group({
    empleadoestudio_id: [''],
    nivelestudio_id: ['', Validators.compose( [Validators.required])],
    carrera: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    semestre_terminado: ['', Validators.compose( [Validators.required, Validators.max(20), Validators.min(0)])],
    fecha_inicio: ['', Validators.compose( [Validators.required])],
    fecha_final: [''],
    en_curso: ['', Validators.compose( [Validators.required])]
  });

  getClasificadorByType(): void {
    this.loading = true;
    this.empleadoService.getClasificadorByType('NEST').subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.tipoEstudio = data.dataArray;
      }
    });
  }

  ngOnInit(): void {
    this.getData();
    this.getClasificadorByType();
  }

  getData(): void {
    const dataObj: object = {
      limit: this.pageSize,
      page: this.pageNumber,
      sort: this.sort,
      filter: this.filterListaEstudio.value,
      order: this.order
    };
    this.loading = true;
    this.empleadoService.getEstudiosEmpleado(dataObj).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        console.log(this.estudiosLista);
        this.estudiosLista = [...data.dataArray];
        this.record = data.dataObjet.records;
        console.log(this.estudiosLista);
        console.log([...data.dataArray]);
        console.log(this.record);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  saveEstudio(): void {
    this.submittedForm = true;
    if (!this.EmpleadoEstudio.valid) {
      return;
    }
    if (this.accion === 'ADD') {
      this.empleadoService.addEmpleadoEstudio(this.EmpleadoEstudio.value).subscribe(data => {
        this.saveEstudioResponse(data);
      }, error => {
        this.router.navigate(['/error']);
      });
    } else if (this.accion === 'EDIT') {
      this.empleadoService.updateEstudioEmpleado(this.EmpleadoEstudio.value).subscribe(data => {
        this.saveEstudioResponse(data);
      }, error => {
        this.router.navigate(['/error']);
      });
    } else {
      this.router.navigate(['/error']);
    }
  }

  saveEstudioResponse(data: any): void {
    this.loading = false;
    this.response = data;
    if (data.state === 200) {
      this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
      this.EmpleadoEstudio.reset();
      this.submittedForm = false;
      this.accion = 'ADD';
      this.textSpan = 'Agregar hijo';
      this.getData();
    } else if (data.state === 402) {
      Swal.fire({
        icon: 'info',
        title: 'Advertencia',
        timer: 1500,
        showConfirmButton: false,
        text: data.msg,
      }).then(() => {
        if (this.logged) {
          this.authService.logout().subscribe(datalog => {
            this.response = datalog;
            if (this.response.state === 200) {
              localStorage.removeItem('logged');
              this.store.dispatch(new fromRoot.CleanRootScope());
              window.location.href = '/login';
            } else if (this.response.state === 500) {
              this.router.navigate(['/error']);
            }
          }, error => {
            this.router.navigate(['/error']);
          });
        }
        this.router.navigate(['/home']);
      });
    } else if (data.state === 401) {
      Swal.fire({
        icon: 'info',
        title: 'Advertencia',
        timer: 1500,
        showConfirmButton: false,
        text: data.msg,
      }).then(() => {
        this.router.navigate(['/home']);
      });
    } else if (data.state === 500) {
      this.router.navigate(['/error']);
    } else {
      this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
    }
  }

  openDataEdit(estudio: Estudio): void {
    console.log(estudio);
    this.textSpan = 'Editar carrera N° ' + estudio.empleadoestudio_id;
    this.accion = 'EDIT';

    this.fechaInicio = new Date(estudio.fecha_inicio);
    this.fechaFinal = new Date(estudio.fecha_final);
    this.EmpleadoEstudio.controls.nivelestudio_id.setValue(estudio.nivelestudio_id);
    this.EmpleadoEstudio.controls.carrera.setValue(estudio.carrera);
    this.EmpleadoEstudio.controls.semestre_terminado.setValue(estudio.semestre_terminado);
    this.EmpleadoEstudio.controls.en_curso.setValue(estudio.en_curso);
    this.EmpleadoEstudio.controls.empleadoestudio_id.setValue(estudio.empleadoestudio_id);
  }

}
