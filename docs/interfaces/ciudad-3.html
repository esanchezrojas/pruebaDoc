    getData(): void {
        const dataObj: object = {
            limit: this.pageSize,
            page: this.pageNumber,
            sort: this.sort,
            filter: this.filterMenuForm.value,
            order: this.order
        };
        this.loading = true;
        this.menuService.getMenuData(dataObj).subscribe(data => {
            this.loading = false;
            if (data.state === 200) {
                this.menus = [...data.dataArray];
                this.record = data.dataObjet.records;
            } else if (data.state === 402) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    if (this.logged) {
                        this.authService.logout().subscribe(datalog => {
                            this.response = datalog;
                            if (this.response.state === 200) {
                                localStorage.removeItem('logged');
                                this.store.dispatch(new fromRoot.CleanRootScope());
                                window.location.href = '/login';
                            } else if (this.response.state === 500) {
                                this.router.navigate(['/error']);
                            }
                        }, error => {
                            this.router.navigate(['/error']);
                        });
                    }
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 401) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 500) {
                this.router.navigate(['/error']);
            } else {
                this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
            }
        }, error => {
            this.router.navigate(['/error']);
        });
    }

    hiddenCam(): void {
        this.hiddenActual = false;
        const datoCambio = this.EmpleadoNuevo.controls.estudia_actual.value;

        if (datoCambio == 'S') {
            this.hiddenActual = true;
            this.EmpleadoNuevo.controls.nivelestudioactual_id.setValidators([Validators.required]);
            this.EmpleadoNuevo.controls.carrera_actual.setValidators([Validators.required]);
            this.EmpleadoNuevo.controls.semestre.setValidators([Validators.required]);
        } else {
            this.hiddenActual = false;
            this.EmpleadoNuevo.controls.nivelestudioactual_id.clearValidators();
            this.EmpleadoNuevo.controls.carrera_actual.clearValidators();
            this.EmpleadoNuevo.controls.semestre.clearValidators();
        }
    }

    changeFecha(): void {
        var d = new Date(this.EmpleadoNuevo.controls.fecha_examed.value);
        var year = d.getFullYear();
        var month = d.getMonth();
        var day = d.getDate();
        this.fecha_exam_new = new Date(year + 2, month, day);
        this.EmpleadoNuevo.controls.fecha_prox_examed.setValue(this.fecha_exam_new);
    }

                    this.router.navigate(['/home']);
                });
            } else if (data.state === 500) {
                this.router.navigate(['/error']);
            } else {
                this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
            }
        }, error => {
            this.router.navigate(['/error']);
        });
    }
