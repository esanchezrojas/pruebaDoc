  addNewPub(): void {
    console.log(this.FormPublicVac.valid);
    this.submittedMenu = true;
    if (!this.FormPublicVac.valid) {
      this.messageService.add({
        severity: 'error',
        summary: 'Advertencia',
        detail: 'Revisa el formulario, todos los campos son obligatorios',
        life: 5000
      });
      return;
    }

    if (this.respon.length === 0) {
      this.messageService.add({severity: 'error', summary: 'Advertencia', detail: 'Ingresa las responsabilidades del cargo', life: 5000});
      return;
    }

    if (this.conoci.length === 0) {
      this.messageService.add({severity: 'error', summary: 'Advertencia', detail: 'Ingresa los conocimientos requeridos', life: 5000});
      return;
    }

    if (this.benefi.length === 0) {
      this.messageService.add({severity: 'error', summary: 'Advertencia', detail: 'Ingresa los beneficios del cargo', life: 5000});
      return;
    }

    this.rePersonalService.addNewPubli(this.FormPublicVac.value, this.respon, this.conoci, this.benefi).subscribe(data => {
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.submittedMenu = false;
        this.FormPublicVac.reset();
        setTimeout(() => {
          this.router.navigate(['/jobs/publicaciones']);
        }, 4000);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      // this.router.navigate(['/erro']);
    });
  }
