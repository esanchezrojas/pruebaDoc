import {Injectable} from '@angular/core';
import {HttpClient, HttpErrorResponse} from "@angular/common/http";
import {throwError} from "rxjs";
import {environment} from "@environments/environment";
import {TreeNode} from "primeng/api";
import {ResponseModel} from "@core/models/response-model/ResponseModel";
import {catchError} from "rxjs/operators";

export interface RePersonal {
  reqpersonal_id: number;
  area_id: number;
  solicitante_id: number;
  cargo_req: number;
  cantidad: number;
  fecha_crea: string;
  profesion: string;
  perfil: string;
  compensacion: number;
  tipocontrato_id: number;
  tiempo: number;
  experiencia_min: string;
  estado: string;
}

@Injectable({
  providedIn: 'root'
})

export class RePersonalService {

  static handleError(error: HttpErrorResponse): any {
    console.log(error);
    return throwError('Ha ocurrido un error..');
  }

  constructor(private http: HttpClient) {
  }

  getRequerimiento(Filtro: any, idReq: string): any {
    const dataObj: object = {
      metodo: 'getRequerimiento',
      data: Filtro,
      idReq: idReq
    };
    return this.http.post<any>(`${environment.url_api}/rePersonalCtrl.php`, dataObj).toPromise().then(res => res as TreeNode[]);
  }

  getVacante(Filtro: any, idReq: string): any {
    const dataObj: object = {
      metodo: 'getVacante',
      data: Filtro,
      idReq: idReq
    };
    return this.http.post<any>(`${environment.url_api}/rePersonalCtrl.php`, dataObj).toPromise().then(res => res as TreeNode[]);
  }

  getPostulados(idVac: string): any {
    const dataObj: object = {
      metodo: 'getPostulados',
      data: idVac
    };
    return this.http.post<any>(`${environment.url_api}/rePersonalCtrl.php`, dataObj).toPromise().then(res => res as TreeNode[]);
  }

  getMyRequerimiento(Filtro: any, idReq: string): any {
    const dataObj: object = {
      metodo: 'getMyRequerimiento',
      data: Filtro,
      idReq: idReq
    };
    return this.http.post<any>(`${environment.url_api}/rePersonalCtrl.php`, dataObj).toPromise().then(res => <TreeNode[]>res.dataObjet);
  }

  addNewRe(rePersonal: RePersonal, respon: any, conoci: any): any {
    const data: object = {
      metodo: 'addNewRe',
      data: rePersonal,
      res: respon,
      con: conoci
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/rePersonalCtrl.php`, data).pipe(
      catchError(RePersonalService.handleError)
    );
  }

  addNewPubli(rePersonal: RePersonal, respon: any, conoci: any, benefi: any): any {
    const data: object = {
      metodo: 'addNewPubli',
      data: rePersonal,
      res: respon,
      con: conoci,
      ben: benefi
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/rePersonalCtrl.php`, data).pipe(
      catchError(RePersonalService.handleError)
    );
  }

  addNewObser(reObser: any): any {
    const data: object = {
      metodo: 'addNewObser',
      data: reObser
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/rePersonalCtrl.php`, data).pipe(
      catchError(RePersonalService.handleError)
    );
  }

  addNewObserVac(reObser: any): any {
    const data: object = {
      metodo: 'addNewObserVac',
      data: reObser
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/rePersonalCtrl.php`, data).pipe(
      catchError(RePersonalService.handleError)
    );
  }

  EditDataRe(rePersonal: RePersonal): any {
    const data: object = {
      metodo: 'EditDataRe',
      data: rePersonal
    }
    return this.http.post<ResponseModel>(`${environment.url_api}/rePersonalCtrl.php`, data).pipe(
      catchError(RePersonalService.handleError)
    );
  }

  changeEstadoVac(data: object): any {
    return this.http.post<ResponseModel>(`${environment.url_api}/rePersonalCtrl.php`, data).pipe(
      catchError(RePersonalService.handleError)
    );
  }

  getImgJobs(): any {
    const data: object = {
      metodo: 'getImgJobs'
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/rePersonalCtrl.php`, data).pipe(
      catchError(RePersonalService.handleError)
    );
  }
}
