    openNew(): void {
        this.menuForm.controls.menu_id.setValue(null);
        this.menuForm.controls.descripcion.setValue('');
        this.menuForm.controls.pagina.setValue('');
        this.menuForm.controls.orden.setValue('');

    hideDialog(): void {
        this.productDialog = false;
        this.editMenuDialog = false;
        this.submitted = false;

    saveMenu(): void {
        this.loading = true;
        this.menuService.addMenu(this.menuForm.value).subscribe(data => {
            this.loading = false;
            this.response = data;
            if (data.state === 200) {
                this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
                this.hideDialog();
                this.getData();
            } else if (data.state === 402) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    if (this.logged) {
                        this.authService.logout().subscribe(datalog => {
                            this.response = datalog;
                            if (this.response.state === 200) {
                                localStorage.removeItem('logged');
                                this.store.dispatch(new fromRoot.CleanRootScope());
                                window.location.href = '/login';
                            } else if (this.response.state === 500) {
                                this.router.navigate(['/error']);
                            }
                        }, error => {
                            this.router.navigate(['/error']);
                        });
                    }
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 401) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 500) {
                this.router.navigate(['/error']);

                this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
            }
        }, error => {
            this.router.navigate(['/error']);
        });
    }

    sortData(e: LazyLoadEvent ): void {
        this.sort = e.sortField;
        this.order = e.sortOrder;
        const num = 100;
        let i = 0;
        while (i < num) {
            if (e.rows * i === e.first) {
                this.pageNumber = i + 1;
            }
            i++;
        }
        this.pageSize = e.rows;
        this.getData();
    }

    filterInput(): void {
        this.getData();
    }

    getData(): void {
        const dataObj: object = {
            limit: this.pageSize,
            page: this.pageNumber,
            sort: this.sort,
            filter: this.filterMenuForm.value,
            order: this.order
        };
        this.loading = true;
        this.menuService.getMenuData(dataObj).subscribe(data => {
            this.loading = false;
            if (data.state === 200) {
