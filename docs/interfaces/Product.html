  constructor(private perfilesServices: PerfilesService,
              private fb: FormBuilder,
              private cb: ChangeDetectorRef,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private router: Router,
              private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private primengConfig: PrimeNGConfig) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  filterNavegacionTree = this.fb.group({
    proceso: ['']
  })

  getClasificadorByType(): void {
    this.perfilesServices.getSelectProcesos().subscribe(data => {
      if (data.state === 200) {
        this.selectProceso =  data.dataArray;
      }
    });
  }

  files: any;

  dataTree(): void {
    this.checkDataAcordListPerfil = true;
    this.checkDataPickPerfil = true;
    this.sourceProducts = [];
    this.targetProducts = [];
    this.perfilesServices.getProcesoApp(this.filterNavegacionTree.value).subscribe(data => {
      this.checkDataAcordListPerfil = false;
      this.files = data.dataObjet.data;
    });
  }

  ngOnInit(): void {
    this.sourceProducts = this.DataTemp.data;
    this.targetProducts = [];
    this.primengConfig.ripple = true;

    this.getClasificadorByType();
  }

  getCargoPerfil(Data: any): void {
    this.checkDataPickPerfil = false;
    this.perfilesServices.getCargosPick(Data.perfilapp_id).subscribe(data => {
      this.checkDataPickPerfil = false;
      this.sourceProducts = data.dataObjet.source;
      this.targetProducts = data.dataObjet.target;
      this.nameSelectedApp = Data.aplicacion + ' - ' + Data.descripcion;
    });
  }

  changeTarget(Event: any, tipo: string): void {
    this.perfilesServices.editTargetCargo(Event.items, tipo).subscribe(data => {
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      // this.router.navigate(['/error']);
    });
  }

}
