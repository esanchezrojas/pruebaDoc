import {Injectable} from '@angular/core';
import {HttpClient, HttpErrorResponse} from '@angular/common/http';
import {environment} from '@environments/environment';
import {throwError} from 'rxjs';
import {catchError} from 'rxjs/operators';
import {ResponseModel} from '@core/models/response-model/ResponseModel';

interface MaestroConfiguracion {
  // Opciones generales
  email1: string;
  email2: string;
  email3: string;
  usulistabase_id: number;
  tiempo_ntc: number;
  version_tyc: number;
  // Impresin
  ticket_pie1: string;
  ticket_pie2: string;
  ticket_pie3: string;
  ticket_pie4: string;
  ticket_pie5: string;
  version_tyc_aliados: number;
  // Recargas
  recarga_pie1: string;
  recarga_pie2: string;
  recarga_pie3: string;
  recarga_pie4: string;
  aplica_doc_recarga: string;
  // Recargas online
  rec_min_pasarela: number;
  rec_min_pasarela_baloto: number;
  // Programa afiliados
  porcen_juego: number;
  porcen_afiliacion: number;
  // Programa referidos
  porcen_referido: number;
  modo_referido: number;
  llave: string;
  // Notas de retiro
  cant_max_notas_ret_x_dia_usu: number;
  val_max_notas_ret_cobra_x_dia_usu: number;
  val_max_una_nota_ret: number;
  tiempo_min_pago_notas_ret_horas: number;
  tiempo_max_pago_notas_ret_horas: number;
  requiere_aprobacion: number;
  porcen_pago_nr: number;
  porcen_retencion: number;
  porcen_iva: number;
  porcen_retrecarga: number;
  retiro_min_bancario: number;
  // Limites de deposito
  limitedep_diario: number;
  limitedep_semanal: number;
  limitedep_mensual: number;
  // Configuracin servicios PTM
  estado: number;
  porcentaje: number;
  rec_minima_movil: string;
  rec_maxima_movil: string;
  // Configuracin Tope promocional
  tope_promo: number;
  // Configuracin casinos
  tiempoExclusionCasino: number;
  // Configuracin Topes APIs
  valormin_recarga_api: number;
  valormax_recarga_api: number;
  valormin_pago_api: number;
  valormax_pago_api: number;
}

@Injectable({
  providedIn: 'root'
})

export class MaestroConfiguracionService {
  static handleError(error: HttpErrorResponse): any {
    console.log(error);
    return throwError('Ha ocurrido un error.');
  }

  constructor(private http: HttpClient) {
  }

  listarConfiguracion(): any {
    const dataObj: object = {
      metodo: 'listarConfiguracion'
    };

    return this.http.post<ResponseModel>(`${environment.url_api}/maestroConfiguracionCtrl.php`, dataObj).pipe(
      catchError(MaestroConfiguracionService.handleError));
  }

  listarListas(): any {
    const dataObj: object = {
      metodo: 'listarList'
    };

    return this.http.post<ResponseModel>(`${environment.url_api}/maestroConfiguracionCtrl.php`, dataObj).pipe(
      catchError(MaestroConfiguracionService.handleError));
  }

  guardarConfiguracion(configuracion: MaestroConfiguracion): any {
    const dataObj: object = {
      metodo: 'guardarConfiguracion',
      filter: configuracion
    };
    return this.http.post<ResponseModel>(`${environment.url_api}/maestroConfiguracionCtrl.php`, dataObj).pipe(
      catchError(MaestroConfiguracionService.handleError));
  }

}
