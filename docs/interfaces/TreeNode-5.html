import {ChangeDetectorRef, Component, Input, OnInit} from '@angular/core';
import {ConfirmationService, MessageService} from 'primeng';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {Observable} from 'rxjs';
import {State} from '@core/models/store-model/StoreModel';
import {RePersonalService} from '@core/services/re-personal/re-personal.service';
import {FormBuilder, Validators} from '@angular/forms';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {Router} from '@angular/router';
import {EmpleadosService} from '@core/services/empleados/empleados.service';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';
import {RePersonal} from '../re-agregar/re-agregar.component';

export interface TreeNode {
  data?: any;
  children?: TreeNode[];
  leaf?: boolean;
  expanded?: boolean;
}

@Component({
  selector: 'app-re-gestion',
  templateUrl: './re-gestion.component.html',
  styleUrls: ['./re-gestion.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class ReGestionComponent implements OnInit {

  constructor(private rePersonalService: RePersonalService,
              private empleadoService: EmpleadosService,
              private fb: FormBuilder,
              private cb: ChangeDetectorRef,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private router: Router,
              private messageService: MessageService,
              private confirmationService: ConfirmationService) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  @Input() idReq: string;

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };

  dato$: Observable<State>;
  logged: Observable<boolean>;
  submittedMenu = false;

  checkAddData = false;
  nameModalRe = '';
  checkDisableRe = true;
  checkAddObser = false;
  checked = true;
  nameModalObser = '';
  checkIdUrl = false;
  idReqText = '';

  typeModal = '';
  selectArea = [];
  selectTipoCon = [];
  selectCargo = [];

  cargoReq = '';
  areaReq = '';
  cantidadReq = 0;
  fechaCreaCeq = '';
  empleadoReq = '';

  FormRePersonal = this.fb.group({
    reqpersonal_id: [{value: '', disabled: false}],
    area_id: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    cargo_req: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    cantidad: [{value: '', disabled: false}, Validators.compose([Validators.required, Validators.max(99)])],
    profesion: [{value: '', disabled: false}, Validators.compose([Validators.required, Validators.maxLength(99)])],
    perfil: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    proposito: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    compensacion: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    tipocontrato_id: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    tiempo: [{value: '', disabled: false}, Validators.compose([Validators.max(99)])],
    experiencia_min: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    ciudad: [{value: '', disabled: false}, Validators.compose([Validators.required])]
  });

  FormRePersonalObser = this.fb.group({
    reqpersonal_id: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    observ: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    estado: [{value: '', disabled: false}, Validators.compose([Validators.required])],
    email: [{value: true, disabled: false}],
    tipo: [{value: 'GE', disabled: false}]
  });

  files: TreeNode[];

  filterNavegacionTree = this.fb.group({
    estado: [''],
    area_id: [''],
    cargo_id: ['']
  });

  getClasificadorByType(): void {
    this.empleadoService.getClasificadorByType('AREA').subscribe(data => {
      if (data.state === 200) {
        this.selectArea = data.dataArray;
      }
    });
    this.empleadoService.getClasificadorByType('TCON').subscribe(data => {
      if (data.state === 200) {
        this.selectTipoCon = data.dataArray;
      }
    });
    this.empleadoService.getCargo().subscribe(data => {
      if (data.state === 200) {
        this.selectCargo = data.dataArray;
      }
    });
  }

  dataTree(data: string): void {
    this.rePersonalService.getRequerimiento(this.filterNavegacionTree.value, data).then(response => {
      this.files = (response.dataObjet.data as TreeNode[]);
      if (this.checkIdUrl) {
        this.openEditData('view', this.files[0].data);
      }
    });
  }

  cargaDataIni(): void {
    let data = '';
    if (this.idReq === undefined) {
      this.idReq = '0';
    } else {
      this.checkIdUrl = true;
      data = atob(decodeURI(this.idReq));
      data = data.slice(10, -10);
      this.idReqText = data;
    }
    this.dataTree(this.idReq);
  }

  ngOnInit(): void {
    this.getClasificadorByType();
    this.cargaDataIni();
  }

  closeIdUrl(): void {
    this.checkIdUrl = false;
    this.idReq = '0';
  }

  openEditData(tipo: string, rePersonal: RePersonal): void {
    this.FormRePersonal.reset();
    this.typeModal = 'edit';
    if (tipo === 'edit') {
      this.nameModalRe = 'Editar proceso de requerimiento de personal';
      this.enableReForm();
    } else if (tipo === 'view') {
      this.nameModalRe = 'Proceso de requerimiento de personal';
      this.disableReForm();
    }
    this.checkAddData = true;

    this.FormRePersonal.controls.area_id.setValue(rePersonal.area_id);
    this.FormRePersonal.controls.reqpersonal_id.setValue(rePersonal.reqpersonal_id);
    this.FormRePersonal.controls.cargo_req.setValue(rePersonal.cargo_req);
    this.FormRePersonal.controls.cantidad.setValue(rePersonal.cantidad);
    this.FormRePersonal.controls.profesion.setValue(rePersonal.profesion);
    this.FormRePersonal.controls.perfil.setValue(rePersonal.descripcion);
    this.FormRePersonal.controls.proposito.setValue(rePersonal.proposito);
    this.FormRePersonal.controls.compensacion.setValue(rePersonal.compensacion);
    this.FormRePersonal.controls.tipocontrato_id.setValue(rePersonal.tipocontrato_id);
    this.FormRePersonal.controls.tiempo.setValue(rePersonal.tiempo);
    this.FormRePersonal.controls.experiencia_min.setValue(rePersonal.experiencia_min);
  }

  openAddObser(rePersonal: RePersonal): void {
    this.checkAddObser = true;
    this.nameModalObser = 'Crear observaci√≥n';

    this.FormRePersonalObser.controls.reqpersonal_id.setValue(rePersonal.reqpersonal_id);
    this.FormRePersonalObser.controls.observ.setValue('');
    this.FormRePersonalObser.controls.email.setValue(true);
    this.FormRePersonalObser.controls.estado.setValue(rePersonal.estado);
    this.checked = true;
    this.cargoReq = rePersonal.cargo;
    this.areaReq = rePersonal.area;
    this.cantidadReq = rePersonal.cantidad;
    this.fechaCreaCeq = rePersonal.fecha_crea;
    this.empleadoReq = rePersonal.nombre;
  }

  addNewObser(): void {
    this.submittedMenu = true;
    if (!this.FormRePersonalObser.valid) {
      return;
    }
    this.rePersonalService.addNewObser(this.FormRePersonalObser.value).subscribe(data => {
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.submittedMenu = false;
        this.FormRePersonal.reset();
        this.hideDialog();
        this.dataTree(this.idReq);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      // this.router.navigate(['/error']);
    });
  }

  disableReForm(): void {
    this.FormRePersonal.controls.area_id.disable();
    this.FormRePersonal.controls.cargo_req.disable();
    this.FormRePersonal.controls.cantidad.disable();
    this.FormRePersonal.controls.profesion.disable();
    this.FormRePersonal.controls.perfil.disable();
    this.FormRePersonal.controls.proposito.disable();
    this.FormRePersonal.controls.compensacion.disable();
    this.FormRePersonal.controls.tipocontrato_id.disable();
    this.FormRePersonal.controls.tiempo.disable();
    this.FormRePersonal.controls.experiencia_min.disable();
  }

  enableReForm(): void {
    this.FormRePersonal.controls.area_id.enable();
    this.FormRePersonal.controls.cargo_req.enable();
    this.FormRePersonal.controls.cantidad.enable();
    this.FormRePersonal.controls.profesion.enable();
    this.FormRePersonal.controls.perfil.enable();
    this.FormRePersonal.controls.proposito.enable();
    this.FormRePersonal.controls.compensacion.enable();
    this.FormRePersonal.controls.tipocontrato_id.enable();
    this.FormRePersonal.controls.tiempo.enable();
    this.FormRePersonal.controls.experiencia_min.enable();
  }

  hideDialog(): void {
    this.checkAddData = false;
    this.checkDisableRe = false;
    this.checkAddObser = false;
  }

  publicReqData(rePersonal: RePersonal): void {
    this.router.navigate(['/jobs/publicaciones/' + rePersonal.reqpersonal_id]);
  }

}
