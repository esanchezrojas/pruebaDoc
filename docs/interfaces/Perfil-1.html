import {AfterViewInit, ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {ConfirmationService, LazyLoadEvent, MessageService} from 'primeng';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {Observable} from 'rxjs';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {Router} from '@angular/router';
import {FormBuilder, Validators} from '@angular/forms';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';

interface Perfil {
  perfil_id: string;
  descripcion: string;
}

@Component({
  selector: 'app-perfil',
  templateUrl: './perfil.component.html',
  styleUrls: ['./perfil.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class PerfilComponent implements OnInit, AfterViewInit{
  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };
  record = 0;
  pageSize = 10;
  pageNumber = 1;
  sort: string;
  order: any;
  loading: boolean;
  logged: Observable<boolean>;
  addPerfilDialog = false;
  editPerfilDialog = false;
  perfilselect = [];
  perfiles: Perfil[];
  perfil: Perfil;
  dato$: Observable<State>;

  filterPerfilForm = this.fb.group({
    perfil_id: '',
    descripcion: ''
  });

  perfilForm = this.fb.group({
    perfil_id: ['', Validators.compose([Validators.required, Validators.maxLength(25)])],
    descripcion: ['', Validators.compose([Validators.required, Validators.maxLength(50)])]
  });
  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private cd: ChangeDetectorRef,
              private utilService: UtilService,
              private router: Router,
              private fb: FormBuilder) {
    this.logged = JSON.parse(localStorage.getItem('logged'));
    this.dato$ = store.pipe(select('rootScope'));
  }

  ngOnInit(): void {
    this.selectperfiles();
  }

  ngAfterViewInit(): void {
    this.cd.detectChanges();
  }

  filterInput(): void {
    this.getData();
  }

  selectperfiles(): any {
    this.menuService.getPerfilesSelect().subscribe(data => {
      this.response = data;
      if (data.state === 200) {
        this.perfilselect = this.response.dataArray;
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'info', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  getData(): void {
    const dataObj: object = {
      limit: this.pageSize,
      page: this.pageNumber,
      sort: this.sort,
      filter: this.filterPerfilForm.value,
      order: this.order
    };
    this.loading = true;
    this.menuService.getPerfilData(dataObj).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        this.perfiles = [...data.dataArray];
        this.record = data.dataObjet.records;
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'info', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  sortData(e: LazyLoadEvent): void {
    this.sort = e.sortField;
    this.order = e.sortOrder;
    const num = 100;
    let i = 0;
    while (i < num) {
      if (e.rows * i === e.first) {
        this.pageNumber = i + 1;
      }
      i++;
    }
    this.pageSize = e.rows;
    this.getData();
  }

  exportExcel(): void {
    this.utilService.exportExcel(this.perfiles, 'perfil');
  }


  addNewPerfil(): void {
    this.loading = true;
    this.menuService.addPerfil(this.perfilForm.value).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});

      }

      this.router.navigate(['/error']);
    });
  }

  hideDialog(): void {
    this.addPerfilDialog = false;
    this.editPerfilDialog = false;
  }

  openNewPerfil(): void {
    this.perfilForm.controls.perfil_id.setValue('');
    this.perfilForm.controls.descripcion.setValue('');
    this.addPerfilDialog = true;
  }

  openEditPerfil(perfil: Perfil): void {
    this.perfilForm.controls.perfil_id.setValue(perfil.perfil_id);
    this.perfilForm.controls.descripcion.setValue(perfil.descripcion);
    this.editPerfilDialog = true;
  }

  deletePerfil(perfil: Perfil): void{
    this.confirmationService.confirm({
      message: `Estas seguro de eliminar el registro ${perfil.descripcion}`,
      header: 'Confirm',
      icon: 'pi pi-exclamation-triangle',
      acceptLabel: 'Si',
      accept: () => {
        this.loading = true;
        this.menuService.deletePerfil(perfil).subscribe(data => {
          this.loading = false;
          if (data.state === 200) {
            this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
            this.hideDialog();

    });
  }
