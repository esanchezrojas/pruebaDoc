import {Injectable} from '@angular/core';
import {CanActivate, ActivatedRouteSnapshot, RouterStateSnapshot, UrlTree, Router} from '@angular/router';
import {Observable, BehaviorSubject} from 'rxjs';
import {AuthService} from '@core/services/auth/auth.service';
import {Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import * as fromRoot from '../rootScopeAction';
import Swal from 'sweetalert2';

@Injectable({
  providedIn: 'root'
})
export class CheckSessionGuard implements CanActivate {
  private route: BehaviorSubject<string> = new BehaviorSubject<string>('');
  public currentRoute: Observable<string>;
  ruta: any;
  public logged: Observable<any>;

  constructor(private router: Router,
              private store: Store<{ rootScope: State }>,
              private authService: AuthService) {
    this.currentRoute = this.route.asObservable();
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  canActivate(
    next: ActivatedRouteSnapshot,
    state: RouterStateSnapshot): Observable<boolean | UrlTree> | Promise<boolean | UrlTree> | boolean | UrlTree {
    if (this.logged === null) {
      window.location.href = '/login';
    }
    setTimeout(() => {
      const route = this.router.url.split('/');
      const routedata = route[1];
      this.route.next(routedata);
      this.currentRoute.subscribe(data => {
        this.ruta = data;
      });
      this.authService.checkSession(this.ruta).subscribe(data => {
        if (data.state === 200) {
          if (data.dataArray.logueado) {
            if (data.dataArray.permisos.pagina === '/login') {
              this.router.navigate(['/home']);
              return;
            }
            if (data.dataArray.permisos.pagina !== undefined) {
              this.store.dispatch(new fromRoot.PermitsRootScope(data.dataArray.permisos));
              this.store.dispatch(new fromRoot.LoggedRootScope(data.dataArray.logueado));
              this.store.dispatch(new fromRoot.NameRootScope(data.dataArray.nombre));
              this.store.dispatch(new fromRoot.ProfileRootScope(data.dataArray.perfil));
              this.store.dispatch(new fromRoot.ProfileTypeRootScope(data.dataArray.perfil_tipo));
              this.store.dispatch(new fromRoot.UsuarioRootScope(data.dataArray.usuario));
              this.store.dispatch(new fromRoot.TokenRootScope(data.dataArray.token));
              this.store.dispatch(new fromRoot.CupoRootScope(data.dataArray.cupo));
              this.store.dispatch(new fromRoot.IpRootScope(data.dataArray.ip));
              return true;
            } else {
              Swal.fire({
                icon: 'info',
                title: 'Advertencia',
                timer: 1000,
                showConfirmButton: false,
                text: 'Validar permisos de ingreso',
              }).then(() => {
                this.router.navigate(['/home']);
              });
              return false;
            }
          } else {
            this.store.dispatch(new fromRoot.CleanRootScope());
            window.location.href = '/login';
          }
        } else if (data.state === 400) {
          localStorage.removeItem('logged');
          localStorage.removeItem('tok');
          this.store.dispatch(new fromRoot.CleanRootScope());
          window.location.href = '/login';
        } else if (data.state === 500) {
          localStorage.removeItem('logged');
          localStorage.removeItem('tok');
          this.store.dispatch(new fromRoot.CleanRootScope());
          this.router.navigate(['/error']);
        }
      }, error => {
        localStorage.removeItem('logged');
        localStorage.removeItem('tok');
        this.store.dispatch(new fromRoot.CleanRootScope());
        this.router.navigate(['/error']);
      });
    }, 5);
    return true;
  }
}
