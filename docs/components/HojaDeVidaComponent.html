import { Component, OnInit } from '@angular/core';
import { ProductoService} from '@core/services/producto/producto.service';

@Component({
  selector: 'app-hoja-de-vida',
  templateUrl: './hoja-de-vida.component.html',
  styleUrls: ['./hoja-de-vida.component.scss']
})
export class HojaDeVidaComponent implements OnInit {

  constructor( private  productoService: ProductoService ) { }
  activo = [];
  loading: boolean;

  ngOnInit(): void {
    this.getData(5);
  }

  getData(id): any {
    this.productoService.getHDV(id).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.activo = data.dataArray;
      }
    });
  }

}
