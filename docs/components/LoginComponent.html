  submit(): void {
    this.authService.login(this.formLogin.value).subscribe(data => {
      this.response = data;
      const Toast = Swal.mixin({
        toast: true,
        position: 'top-end',
        showConfirmButton: false,
        timer: 1500,
        timerProgressBar: true,
        onOpen: (toast) => {
          toast.addEventListener('mouseenter', Swal.stopTimer);
          toast.addEventListener('mouseleave', Swal.resumeTimer);
        }
      });
      if (this.response.state === 200) {
        this.store.dispatch(new fromRoot.LoggedRootScope(data.dataObjet.logueado));
        this.store.dispatch(new fromRoot.NameRootScope(data.dataObjet.nombre));
        this.store.dispatch(new fromRoot.ProfileRootScope(data.dataObjet.perfil));
        this.store.dispatch(new fromRoot.ProfileTypeRootScope(data.dataObjet.perfil_tipo));
        this.store.dispatch(new fromRoot.UsuarioRootScope(data.dataObjet.usuario));
        this.store.dispatch(new fromRoot.TokenRootScope(data.dataObjet.token));
        this.store.dispatch(new fromRoot.IpRootScope(data.dataObjet.ip));
        localStorage.setItem('tok', JSON.stringify(data.dataObjet.token));
        localStorage.setItem('logged', JSON.stringify(true));
        Swal.fire({
          icon: 'success',
          title: 'Bienvenido',
          timer: 1500,
          showConfirmButton: false,
          text: 'Usuario logueado corretamente',
        }).then(() => {
          window.location.href = '/home';
        });

      } else if (this.response.state === 500) {
        this.router.navigate(['/error']);
      } else if (this.response.state === 402) {
        localStorage.removeItem('logged');
        localStorage.removeItem('tok');
        this.store.dispatch(new fromRoot.CleanRootScope());
        Toast.fire({
          icon: 'info',
          title: 'Se ha detectado un nuevo ingreso a tu cuenta'
        });
      } else if (this.response.state === 403) {
        this.reload();
        Swal.fire({
          title: 'Señor usuario',
          input: 'text',
          text: 'Por favor ingrese el token:',
          inputAttributes: {
            autocapitalize: 'off'
          },
          showCancelButton: true,
          confirmButtonText: 'OK',
          showLoaderOnConfirm: true,
        }).then((result) => {
          if (result.isConfirmed) {
            this.formLogin.controls.verify.setValue(result.value);
            this.authService.login(this.formLogin.value).subscribe(datos => {
              this.response = datos;
              if (this.response.state === 200) {
                this.store.dispatch(new fromRoot.LoggedRootScope(data.dataObjet.logueado));
                this.store.dispatch(new fromRoot.NameRootScope(data.dataObjet.nombre));
                this.store.dispatch(new fromRoot.ProfileRootScope(data.dataObjet.perfil));
                this.store.dispatch(new fromRoot.ProfileTypeRootScope(data.dataObjet.perfil_tipo));
                this.store.dispatch(new fromRoot.UsuarioRootScope(data.dataObjet.usuario));
                this.store.dispatch(new fromRoot.TokenRootScope(data.dataObjet.token));
                this.store.dispatch(new fromRoot.IpRootScope(data.dataObjet.ip));
                localStorage.setItem('tok', JSON.stringify(data.dataObjet.token));
                localStorage.setItem('logged', JSON.stringify(true));
                Swal.fire({
                  icon: 'success',
                  title: 'Bienvenido',
                  timer: 1500,
                  showConfirmButton: false,
                  text: 'Usuario logueado corretamente',
                }).then(() => {
                  window.location.href = '/home';
                });
              } else if (this.response.state === 500) {
                this.formLogin.controls.verify.setValue(null);
                this.router.navigate(['/error']);
              } else if (this.response.state === 402) {
                this.formLogin.controls.verify.setValue(null);
                localStorage.removeItem('logged');
                localStorage.removeItem('tok');
                this.store.dispatch(new fromRoot.CleanRootScope());
                Toast.fire({
                  icon: 'info',
                  title: 'Se ha detectado un nuevo ingreso a tu cuenta'
                });
              } else {
                this.formLogin.controls.verify.setValue(null);
                Swal.fire({
                  icon: 'info',
                  title: 'Advertencia',
                  timer: 1500,
                  showConfirmButton: false,
                  text: this.response.msg,
                }).then(() => {
                  this.reload();
                });
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
        });
      } else {
        Swal.fire({
          icon: 'error',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: this.response.msg,
        }).then(() => {
          this.reload();
        });
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }
