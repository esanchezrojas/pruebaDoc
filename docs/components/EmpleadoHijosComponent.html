import {AfterViewInit, ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {Observable} from 'rxjs';
import {State} from '@core/models/store-model/StoreModel';
import {FormBuilder, Validators} from '@angular/forms';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {Router} from '@angular/router';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';
import {EmpleadosService} from '@core/services/empleados/empleados.service';
import {ConfirmationService, LazyLoadEvent, MessageService} from 'primeng';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';

interface Hijo {
  empleadohijo_id: number;
  empleado_id: number;
  nombre_hijo: string;
  fecha_nacim: string;
  nivelestudio_id: number;
  institucion: string;
  nit_institucion: string;
  tipohijo_id: number;
  estado: string;
}

@Component({
  selector: 'app-empleado-hijos',
  templateUrl: './empleado-hijos.component.html',
  styleUrls: ['./empleado-hijos.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class EmpleadoHijosComponent implements OnInit, AfterViewInit {

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private utilService: UtilService,
              private empleadoService: EmpleadosService,
              private cd: ChangeDetectorRef,
              private router: Router,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };
  dato$: Observable<State>;
  logged: Observable<boolean>;
  loading = true;
  nivelActual = [];
  tipoHijo = [];
  pageSize = 100;
  pageNumber = 1;
  record = 0;
  sort: string;
  order: any;
  hijosLista = [];
  submittedForm = false;
  textSpan = 'Agregar hijo';
  accion = 'ADD';

  fechaNacim = new Date();

  filterListaHijos = this.fb.group({
    tipo: 'Admin'
  });

  EmpleadoHijo = this.fb.group({
    nombre_hijo: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    fecha_nacim: ['', Validators.compose( [Validators.required])],
    nivelestudio_id: ['', Validators.compose( [Validators.required])],
    institucion: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    nit_institucion: ['', Validators.compose([Validators.required, Validators.maxLength(13)])],
    tipohijo_id: ['', Validators.compose( [Validators.required])],
    estado: ['A', Validators.compose( [Validators.required])],
    empleadohijo_id: ['']
  });

  getClasificadorByType(): void {
    this.loading = true;
    this.empleadoService.getClasificadorByType('NEST').subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.nivelActual = data.dataArray;
      }
    });
    this.empleadoService.getClasificadorByType('THIJ').subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.tipoHijo = data.dataArray;
      }
    });
  }

  ngOnInit(): void {
  }

  getData(): void {
    const dataObj: object = {
      limit: this.pageSize,
      page: this.pageNumber,
      sort: this.sort,
      filter: this.filterListaHijos.value,
      order: this.order
    };
    this.loading = true;
    this.empleadoService.getHijosEmpleado(dataObj).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        console.log(this.hijosLista);
        this.hijosLista = [...data.dataArray];
        this.record = data.dataObjet.records;
        console.log(this.hijosLista);
        console.log([...data.dataArray]);
        console.log(this.record);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  ngAfterViewInit(): void {
    this.getClasificadorByType();
    this.getData();
  }

  saveHijo(): void {
    this.submittedForm = true;
    if (!this.EmpleadoHijo.valid) {
      return;
    }
    if (this.accion === 'ADD') {
      this.empleadoService.addEmpleadoHijo(this.EmpleadoHijo.value).subscribe(data => {
        this.saveHijoResponse(data);
      }, error => {
        this.router.navigate(['/error']);
      });
    } else if (this.accion === 'EDIT') {
      this.empleadoService.updateEmpleadoHijo(this.EmpleadoHijo.value).subscribe(data => {
        this.saveHijoResponse(data);
      }, error => {
        this.router.navigate(['/error']);
      });
    } else {
      this.router.navigate(['/error']);
    }
  }

  saveHijoResponse(data: any): void {
    this.loading = false;
    this.response = data;
    if (data.state === 200) {
      this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
      this.EmpleadoHijo.reset();
      this.submittedForm = false;
      this.accion = 'ADD';
      this.textSpan = 'Agregar hijo';
      this.getData();
    } else if (data.state === 402) {
      Swal.fire({
        icon: 'info',
        title: 'Advertencia',
        timer: 1500,
        showConfirmButton: false,
        text: data.msg,
      }).then(() => {
        if (this.logged) {
          this.authService.logout().subscribe(datalog => {
            this.response = datalog;
            if (this.response.state === 200) {
              localStorage.removeItem('logged');
              this.store.dispatch(new fromRoot.CleanRootScope());
              window.location.href = '/login';
            } else if (this.response.state === 500) {
              this.router.navigate(['/error']);
            }
          }, error => {
            this.router.navigate(['/error']);
          });
        }
        this.router.navigate(['/home']);
      });
    } else if (data.state === 401) {
      Swal.fire({
        icon: 'info',
        title: 'Advertencia',
        timer: 1500,
        showConfirmButton: false,
        text: data.msg,
      }).then(() => {
        this.router.navigate(['/home']);
      });
    } else if (data.state === 500) {
      this.router.navigate(['/error']);
    } else {
      this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
    }
  }

  openDataEdit(hijo: Hijo): void {
    console.log(hijo);
    this.textSpan = 'Editar hijo N° ' + hijo.empleadohijo_id;
    this.accion = 'EDIT';

    this.fechaNacim = new Date(hijo.fecha_nacim);
    this.EmpleadoHijo.controls.nombre_hijo.setValue(hijo.nombre_hijo);
    this.EmpleadoHijo.controls.nivelestudio_id.setValue(hijo.nivelestudio_id);
    this.EmpleadoHijo.controls.institucion.setValue(hijo.institucion);
    this.EmpleadoHijo.controls.nit_institucion.setValue(hijo.nit_institucion);
    this.EmpleadoHijo.controls.tipohijo_id.setValue(hijo.tipohijo_id);
    this.EmpleadoHijo.controls.empleadohijo_id.setValue(hijo.empleadohijo_id);
    this.EmpleadoHijo.controls.estado.setValue(hijo.estado);
  }

}
