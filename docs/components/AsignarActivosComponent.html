import {ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {FormBuilder, FormControl, Validators} from '@angular/forms';
import {Observable} from 'rxjs';
import {State} from '@core/models/store-model/StoreModel';
import {ConfirmationService, MessageService} from 'primeng';
import {AuthService} from '@core/services/auth/auth.service';
import {AsignarActivosService} from '@core/services/asignar-activos/asignar-activos.service';
import {select, Store} from '@ngrx/store';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {Router} from '@angular/router';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';

@Component({
  selector: 'app-asignar-activos',
  templateUrl: './asignar-activos.component.html',
  styleUrls: ['./asignar-activos.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class AsignarActivosComponent implements OnInit {
  private response: any;
  formSubmitted = false;
  public logged: Observable<boolean>;
  dato$: Observable<State>;
  pageNumber: number;
  loading: boolean;
  Activos = [];
  Empleados = [];
  AsignarActivos = this.fb.group({
    Empleado: ['', Validators.compose([Validators.required])],
    Activo: ['', Validators.compose([Validators.required])]

  });

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private asignarActivosService: AsignarActivosService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private utilService: UtilService,
              private router: Router,
              private cd: ChangeDetectorRef,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
    this.pageNumber = 1;
  }

  ngOnInit(): void {
    this.getActivos();
    this.getEmpleados();
  }
  getActivos(): any {
    this.asignarActivosService.getActivos().subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.Activos = data.dataArray;
      }
    });
  }
  getEmpleados(): any {
    this.asignarActivosService.getEmpleados().subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.Empleados = data.dataArray;
      }
    });
  }
  SaveInput(): any{
    this.loading = true;
    this.formSubmitted = true;
    if (!this.AsignarActivos.valid) {
      return;
    }
    this.asignarActivosService.saveActivo(this.AsignarActivos.value).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        Swal.fire({
          icon: 'success',
          title: 'Exito',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          window.location.reload();
        });
      }
      else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }
}
