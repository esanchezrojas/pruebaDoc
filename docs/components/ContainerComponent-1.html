  setParam(): void {
    this.actualizaService.getParam(this.ParamUrl).subscribe(data => {

        this.empleadosLista = [...data.dataArray];

        Swal.fire({
          icon: 'error',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: this.response.msg,
        }).then(() => {
          this.router.navigate(['/login']);
        });
      } else {
        Swal.fire({
          icon: 'error',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: this.response.msg,
        }).then(() => {
          this.reload();
        });
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  editaEmpleado(): void {
    this.loading = true;
    this.actualizaService.actualizaEmpleado(this.EmpleadoEdit.value).subscribe(data => {
      this.loading = false;
      this.response = data;

        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.hideDialog();
          this.router.navigate(['/home']);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });

        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {

        });

        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      // this.router.navigate(['/error']);
    });
  }
