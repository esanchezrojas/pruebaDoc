import {Component, OnInit} from '@angular/core';
import {Observable} from 'rxjs';
import {State} from '@core/models/store-model/StoreModel';
import {select, Store} from '@ngrx/store';

@Component({
  selector: 'app-container',
  templateUrl: './container.component.html',
  styleUrls: ['./container.component.scss']
})
export class ContainerComponent implements OnInit {
  tab: number;
  menu = false;
  subMenu = false;
  menuPerfil = false;
  perfil = false;
  dato$: Observable<State>;

  constructor(private store: Store<{ rootScope: State }>) {
    this.dato$ = store.pipe(select('rootScope'));
    this.tab = JSON.parse(localStorage.getItem('tap-menu'));
    if (this.tab === undefined || this.tab === null) {
      this.tab = 0;
    }
    this.initTapMenu(this.tab);
  }

  ngOnInit(): void {
  }

  validIndex(e): void {
   // console.log(e);
  }

  initTapMenu(idMenu: number): void {
    switch (idMenu) {
      case 0:
        this.subMenu = false;
        this.menu = true;
        this.menuPerfil = false;
        this.perfil = false;
        break;
      case 1:
        this.subMenu = true;
        this.menu = false;
        this.menuPerfil = false;
        this.perfil = false;
        break;
      case 2:
        this.subMenu = true;
        this.menu = false;
        this.menuPerfil = true;
        this.perfil = false;
        break;
      case 3:
        this.subMenu = true;
        this.menu = false;
        this.menuPerfil = false;
        this.perfil = true;
        break;
    }
  }

  changeTabMenu(e): void {
    localStorage.setItem('tap-menu', JSON.stringify(e.index));
    this.tab = JSON.parse(localStorage.getItem('tap-menu'));
    this.initTapMenu(this.tab);
  }
}
