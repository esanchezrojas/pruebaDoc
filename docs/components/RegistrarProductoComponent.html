import {ChangeDetectorRef, Component, OnInit, Input } from '@angular/core';
import {ConfirmationService, LazyLoadEvent, MessageService, SortEvent} from 'primeng';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {Router} from '@angular/router';
import {FormBuilder, Validators} from '@angular/forms';
import {Observable} from 'rxjs';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {FormGroup, FormControl, FormArray} from '@angular/forms';
import {stringifyConfiguration} from 'tslint/lib/configuration';
import {ProductoService} from '@core/services/producto/producto.service';
import {ProveedoresService} from '@core/services/proveerdores/proveedores.service';

@Component({
  selector: 'app-registrar-producto',
  templateUrl: './registrar-producto.component.html',
  styleUrls: ['./registrar-producto.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class RegistrarProductoComponent implements OnInit {

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private proveedoresService: ProveedoresService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private utilService: UtilService,
              private router: Router,
              private cd: ChangeDetectorRef,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
    this.pageNumber = 1;
  }
  @Input() idProv: number;
  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };
  submittedForm = false;
  public logged: Observable<boolean>;
  dato$: Observable<State>;
  pageNumber: number;
  loading: boolean;
  cl1 = '';
  selectProducto_id = [];
  ProductoNuevo = this.fb.group({
    producto_id: ['', Validators.compose([Validators.required])]
  });
  ngOnInit(): void {
    this.getProductP();
  }

  SaveInput(): any{

    //console.log(this.ProductoNuevo.value);
    //console.log(this.idProv);

    //return false;

    /*alert(this.idProvN);

    return false;*/

    this.loading = true;
    this.submittedForm = true;
    if (!this.ProductoNuevo.valid) {
      return;
    }
    this.proveedoresService.saveProducto(this.idProv, this.ProductoNuevo.value).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        Swal.fire({
          icon: 'success',
          title: 'Exito',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          window.location.reload();
        });
      }
      else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  getProductP(): void {
    this.proveedoresService.getProducto(this.idProv).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.selectProducto_id = data.dataArray;
      }
    });
  }

}
