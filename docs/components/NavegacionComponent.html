import {ChangeDetectorRef, Component, OnInit } from '@angular/core';
import {PerfilesService} from "@core/services/perfiles/perfiles.service";
import {ConfirmationService, MessageService} from "primeng";
import {AuthService} from "@core/services/auth/auth.service";
import {select, Store} from "@ngrx/store";
import {State} from "@core/models/store-model/StoreModel";
import {FormBuilder, Validators} from "@angular/forms";
import {Observable} from "rxjs";
import * as fromRoot from "../../../rootScopeAction";
import {Router} from "@angular/router";
import {validate} from "codelyzer/walkerFactory/walkerFn";
import Swal from "sweetalert2";
import {ResponseModel} from "@core/models/response-model/ResponseModel";

export interface TreeNode {
  data?: any;
  children?: TreeNode[];
  leaf?: boolean;
  expanded?: boolean;
}

interface Data {
  proceso_id: number;
  aplicacion_id: number;
  menu_id: number;
  id: number;
  descripcion: string;
  estado: string;
  tipo: string;
  ruta: string;
}

@Component({
  selector: 'app-navegacion',
  templateUrl: './navegacion.component.html',
  styleUrls: ['./navegacion.component.scss'],
  providers: [MessageService, ConfirmationService]
})

export class NavegacionComponent implements OnInit {

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };

  dato$: Observable<State>;
  logged: Observable<boolean>

  selectProceso = [];

  constructor(private perfilesServices: PerfilesService,
              private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private fb: FormBuilder,
              private cd: ChangeDetectorRef,
              private router: Router) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  NewDataForm = this.fb.group({
    id: [''],
    crear: ['', Validators.compose([Validators.required])],
    descripcion: ['', Validators.compose([Validators.required])],
    ruta: ['']
  })

  EditDataForm = this.fb.group({
    id: ['', Validators.compose([Validators.required])],
    edit: ['', Validators.compose([Validators.required])],
    descripcion: ['', Validators.compose([Validators.required])],
    ruta: ['']
  })

  filterNavegacionTree = this.fb.group({
    proceso: ['']
  })

  files: TreeNode[];
  newData: boolean = false;
  editData: boolean = false;
  nameModal: string;
  nameModalEdit: string;

  rutaCheck: boolean = true;
  rutaCheckEdit: boolean = true;

  dataTree(): void {
    this.perfilesServices.getProcesos(this.filterNavegacionTree.value).then(data => {
      this.files = <TreeNode[]> data.data;
    });
  }

  ngOnInit() {
    this.dataTree();
    this.getClasificadorByType();
  }

  getClasificadorByType(): void {
    this.perfilesServices.getSelectProcesos().subscribe(data => {
      if (data.state === 200) {
        this.selectProceso =  data.dataArray;
      }
    });
  }

  openAddData(data: Data, crear: string): void {
    this.newData = true;
    this.rutaCheck = true;
    this.nameModal = 'Crear ' + crear + ' en ' + data.descripcion;
    this.NewDataForm.controls.crear.setValue(crear);

    if (crear == 'aplicacion') {
      this.NewDataForm.controls.id.setValue(data.proceso_id);
      console.log(data.proceso_id);
    } else if (crear == 'menu') {
      this.NewDataForm.controls.id.setValue(data.aplicacion_id);
    }
  }

  openEditData(data: Data, editar: string): void {
    this.editData = true;
    this.rutaCheckEdit = true;
    this.nameModalEdit = 'Editar ' + editar + ' ' + data.descripcion;

    this.EditDataForm.controls.edit.setValue(editar);
    this.EditDataForm.controls.ruta.setValue(data.ruta);
    this.EditDataForm.controls.descripcion.setValue(data.descripcion);

    if (editar == 'aplicacion') {
      this.EditDataForm.controls.id.setValue(data.aplicacion_id);
    } else if (editar == 'menu') {
      this.EditDataForm.controls.id.setValue(data.menu_id);
    } else if (editar == 'proceso') {
      this.EditDataForm.controls.id.setValue(data.proceso_id);
      this.rutaCheckEdit = false;
    }
  }

  openAddDataProceso(crear: string): void {
    this.newData = true;
    this.rutaCheck = false;
    this.nameModal = 'Crear proceso nuevo';
    this.NewDataForm.controls.crear.setValue(crear);
  }

  addData(): void {
    this.perfilesServices.addDataNavegacion(this.NewDataForm.value).subscribe(data => {
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.NewDataForm.controls.ruta.setValue('');
        this.NewDataForm.controls.descripcion.setValue('');
        this.NewDataForm.controls.crear.setValue('');
        this.NewDataForm.controls.id.setValue('');
        this.hideDialog();
        this.dataTree();
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      // this.router.navigate(['/error']);
    });
  }

  editEstado(tipo: string, data: Data): void {
    let textTipo = '';
    let textAnt = 'el'

    if (tipo == 'I') {
      textTipo = 'incativar';
    } else if (tipo == 'A') {
      textTipo = 'activar';
    }

    if (data.tipo == 'Aplicacion') {
      textAnt = 'la'
    }

    this.confirmationService.confirm({
      message: `Estas seguro que quieres ${textTipo} ${textAnt} ${data.tipo} ${data.descripcion}`,
      header: 'Confirm',
      icon: 'pi pi-exclamation-triangle',
      acceptLabel: 'Si',
      accept: () => {
        this.perfilesServices.editEstadoNavegacion(tipo, data).subscribe(data => {
          if (data.state === 200) {
            this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
            this.dataTree();
          } else if (data.state === 402) {
            Swal.fire({
              icon: 'info',
              title: 'Advertencia',
              timer: 1500,
              showConfirmButton: false,
              text: data.msg,
            }).then(() => {
              if (this.logged) {
                this.authService.logout().subscribe(datalog => {
                  this.response = datalog;
                  if (this.response.state === 200) {
                    localStorage.removeItem('logged');
                    this.store.dispatch(new fromRoot.CleanRootScope());
                    window.location.href = '/login';
                  } else if (this.response.state === 500) {
                    this.router.navigate(['/error']);
                  }
                }, error => {
                  this.router.navigate(['/error']);
                });
              }
              this.router.navigate(['/home']);
            });
          } else if (data.state === 401) {
            Swal.fire({
              icon: 'info',
              title: 'Advertencia',
              timer: 1500,
              showConfirmButton: false,
              text: data.msg,
            }).then(() => {
              this.router.navigate(['/home']);
            });
          } else if (data.state === 500) {
            this.router.navigate(['/error']);
          } else {
            this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
          }
        }, error => {
          this.router.navigate(['/error']);
        });
      }
    });
  }

  editDataData(): void {
    this.perfilesServices.editDataNavegacion(this.EditDataForm.value).subscribe(data => {
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.hideDialog();
        this.dataTree();
        this.EditDataForm.controls.ruta.setValue('');
        this.EditDataForm.controls.descripcion.setValue('');
        this.EditDataForm.controls.crear.setValue('');
        this.EditDataForm.controls.id.setValue('');
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      // this.router.navigate(['/error']);
    });
  }

  hideDialog(): void {
    this.newData = false;
    this.editData = false;
  }

}
