import {ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {FormBuilder, Validators} from '@angular/forms';
import {ConfirmationService, MessageService} from 'primeng';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {Router} from '@angular/router';
import {Observable} from 'rxjs';
import Swal from 'sweetalert2';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {CambioContrasenaService} from '@core/services/cambio-contrasena/cambio-contrasena.service';

@Component({
  selector: 'app-container',
  templateUrl: './container.component.html',
  styleUrls: ['./container.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class ContainerComponent implements OnInit {
  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };

  public logged: Observable<boolean>;
  dato$: Observable<State>;

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private cambioContrasenaService: CambioContrasenaService,
              private store: Store<{ rootScope: State }>,
              private router: Router,
              private cd: ChangeDetectorRef,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  cambioContrasena = this.fb.group({
    contrasena_actual: ['', Validators.compose([
        Validators.required,
        Validators.minLength(8),
        Validators.maxLength(15)
      ]
    )],
    n_contrasena: ['', Validators.compose([
        Validators.required,
        Validators.minLength(8),
        Validators.maxLength(15),
        Validators.pattern(new RegExp('^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#\$%\^&\*{}-])(?=.{8,30})'))
      ]
    )], c_contrasena: ['', Validators.compose([
        Validators.required,
        Validators.minLength(8),
        Validators.maxLength(15),
        Validators.pattern(new RegExp('^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#\$%\^&\*{}-])(?=.{8,30})'))
      ]
    )],
    metodo: ['guardarConfiguracion']
  });

  onRefresh() {
    this.router.routeReuseStrategy.shouldReuseRoute = function() {
      return false;
    };

    const currentUrl = this.router.url + '?';

    this.router.navigateByUrl(currentUrl)
      .then(() => {
        this.router.navigated = false;
        this.router.navigate([this.router.url]);
      });
  }

  cambioContrasenaSubmit(): void {
    this.cambioContrasenaService.cambioContrasena(this.cambioContrasena.value).subscribe(data => {
      this.response = data;
      if (data.state === 200) {
        Swal.fire({
          icon: 'success',
          title: '',
          timer: 5000,
          showConfirmButton: false,
          text: data.msg,
        });
        this.onRefresh();
      } else {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 4000,
          showConfirmButton: false,
          text: data.msg,
        });
      }
    });
  }

  ngOnInit(): void {
  }

}
