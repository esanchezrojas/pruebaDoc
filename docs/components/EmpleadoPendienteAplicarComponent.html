  ApruebaEmpleado(): void {
    this.loading = true;
    this.empleadoService.ApruebaEmpleado(this.EmpleadoEdit.value).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        setTimeout( function () {
            window.location.reload();
          }, 4000
        )
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
