import {AfterViewInit, ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {ConfirmationService, LazyLoadEvent, MessageService, SortEvent} from 'primeng';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {EmpleadosService} from '@core/services/empleados/empleados.service';
import {Router} from '@angular/router';
import {FormBuilder, Validators} from '@angular/forms';
import {Observable} from 'rxjs';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {FormGroup, FormControl, FormArray} from '@angular/forms';

interface Empleado {
  cedula: string;
  tipo_doc: string;
  fecha_exp: string;
  p_nombre: string;
  s_nombre: string;
  p_apellido: string;
  s_apellido: string;
  cargo: number;
  area_id: number;
}

interface ciudad {
  ciudad_id: number;
  descripcion: string;
}

@Component({
  selector: 'app-empleado-nuevo-corto',
  templateUrl: './empleado-nuevo-corto.component.html',
  styleUrls: ['./empleado-nuevo-corto.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class EmpleadoNuevoCortoComponent implements OnInit, AfterViewInit {

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };

  public logged: Observable<boolean>;
  dato$: Observable<State>;
  pageNumber: number;
  selectArea = [];
  selectCargo = [];
  loading = false;
  submittedForm = false;
  cedulaId: any;
  ciudadExpId: any;
  resultsCity: ciudad[] = [];

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private empleadoService: EmpleadosService,
              private utilService: UtilService,
              private router: Router,
              private cd: ChangeDetectorRef,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
    this.pageNumber = 1;
  }

  EmpleadoNuevoCorto = this.fb.group({
    cedula: ['', Validators.compose([Validators.required, Validators.maxLength(10), Validators.minLength(5)])],
    tipo_doc: ['', Validators.compose([Validators.required])],
    fecha_exp: ['', Validators.compose([Validators.required])],
    lugar_exp: ['', Validators.compose( [Validators.required])],
    p_nombre: ['', Validators.compose([Validators.required, Validators.maxLength(25)])],
    s_nombre: ['', Validators.compose([Validators.maxLength(25)])],
    p_apellido: ['', Validators.compose([Validators.required, Validators.maxLength(25)])],
    s_apellido: ['', Validators.compose([Validators.maxLength(25)])],
    email_corporativo: ['', Validators.compose([Validators.required, Validators.email, Validators.maxLength(150)])],
    cargo: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    area_id: ['', Validators.compose([Validators.required])],
    fecha_contrato: ['', Validators.compose([Validators.required])]
  });

  ngOnInit(): void {
    this.getClasificadorByType();
  }

  searchCity(event): void {
    this.empleadoService.getCiudad(event.query).subscribe(data => {
      this.resultsCity = data.dataArray;
    });
  }

  ngAfterViewInit(): void {
    this.cd.detectChanges();
  }

  getClasificadorByType(): void {
    this.loading = true;
    this.empleadoService.getClasificadorByType('AREA').subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.selectArea = data.dataArray;
      }
    });
    this.empleadoService.getCargo().subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.selectCargo = data.dataArray;
      }
    });
  }

  saveData(): void {
    this.submittedForm = true;
    if (!this.EmpleadoNuevoCorto.valid) {
      return;
    }
    this.loading = true;
    this.empleadoService.addEmpleadoCorto(this.EmpleadoNuevoCorto.value).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        setTimeout(() => {
            window.location.reload();
          }, 4000
        );
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }
}
