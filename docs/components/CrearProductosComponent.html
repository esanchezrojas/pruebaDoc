import {AfterViewInit, ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {ConfirmationService, LazyLoadEvent, MessageService, SortEvent} from 'primeng';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {Router} from '@angular/router';
import {FormBuilder, Validators} from '@angular/forms';
import {Observable} from 'rxjs';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {FormGroup, FormControl, FormArray} from '@angular/forms';
import {stringifyConfiguration} from 'tslint/lib/configuration';
import {ProductoService} from '@core/services/producto/producto.service';

@Component({
  selector: 'app-crear-productos',
  templateUrl: './crear-productos.component.html',
  styleUrls: ['./crear-productos.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class CrearProductosComponent implements OnInit, AfterViewInit {
  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };
  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private  productoService: ProductoService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private utilService: UtilService,
              private router: Router,
              private cd: ChangeDetectorRef,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
    this.pageNumber = 1;
  }
  submittedForm = false;
  public logged: Observable<boolean>;
  dato$: Observable<State>;
  pageNumber: number;
  loading: boolean;
  cl1 = '';
  cl2 = '';
  marcas = [];
  selectclasificador1 = [];
  selectclasificador2 = [];
  selectclasificador3 = [];
  ProductoNuevo = this.fb.group({
    descripcion: ['', Validators.compose([Validators.required, Validators.maxLength(150), Validators.minLength(5)])],
    clasificador1: ['', Validators.compose([Validators.required])],
    clasificador2: ['', Validators.compose([Validators.required])],
    clasificador3: ['', Validators.compose([Validators.required])],
    marca: ['', Validators.compose([Validators.required])],
    modelo: ['', Validators.compose([Validators.maxLength(80)])]

  });
  ngOnInit(): void {
    this.getClasificadorByType();
    this.getMarca();
  }
  SaveInput(): any{
    this.loading = true;
    this.submittedForm = true;
    if (!this.ProductoNuevo.valid) {
      return;
    }
    this.productoService.saveProducto(this.ProductoNuevo.value).subscribe(data => {
      this.loading = false;
      this.response = data;
      if (data.state === 200) {
        Swal.fire({
          icon: 'success',
          title: 'Exito',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          window.location.reload();
        });
      }
       else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }
  getClasificadorByType(): void {
    this.productoService.getClasificadorByType('0').subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.selectclasificador1 = data.dataArray;
      }
    });
  }
  getClasificadorByType2(): void {
    this.productoService.getClasificadorByType(this.cl1).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.selectclasificador2 = data.dataArray;
      }
    });
  }
  getClasificadorByType3(): void {
    this.productoService.getClasificadorByType(this.cl2).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.selectclasificador3 = data.dataArray;
      }
    });
  }
  getMarca(): void {
    this.productoService.getMarca().subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.marcas = data.dataArray;
      }
    });
  }
  ngAfterViewInit(): void {
  }

}
