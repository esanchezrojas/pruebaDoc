import {ChangeDetectorRef, Component, Input, OnInit} from '@angular/core';
import {ResponseModel} from "@core/models/response-model/ResponseModel";
import {Observable} from "rxjs";
import {State} from "@core/models/store-model/StoreModel";
import {ConfirmationService, LazyLoadEvent, MessageService} from "primeng";
import {AuthService} from "@core/services/auth/auth.service";
import {select, Store} from "@ngrx/store";
import {UtilService} from "@core/services/util/util.service";
import {Router} from "@angular/router";
import {FormBuilder, Validators} from "@angular/forms";
import {VacacionesService} from "@core/services/vacaciones/vacaciones.service";
import Swal from "sweetalert2";
import * as fromRoot from "../../../rootScopeAction";

interface Vacaciones {
    solicitud_id: number;
    empleado_id: number;
    dias_disfrute: number;
    dias_dinero: number;
    jefe_id: number;
    fecha_crea: string;
    tipo: string;
    fecha_ini: string;
    fecha_fin: string;
    fecha_aproba_rechazo: string;
    motivo_coment: string;
    observ: string;
    estado: string;
}

@Component({
    selector: 'app-agregar',
    templateUrl: './agregar.component.html',
    styleUrls: ['./agregar.component.scss']
})

export class AgregarComponent implements OnInit {

    @Input() idReq: string;

    response: ResponseModel = {
        dataArray: [],
        msg: '', state: 0
    };

    record = 0;
    display: boolean = false;
    pageSize = 100;
    pageNumber = 1;
    sort: string;
    order: any;
    loading: boolean;
    logged: Observable<boolean>;
    dato$: Observable<State>;
    checkOpenAddSol: boolean = false;
    submittedMenu: boolean = false;

    vacacionesLista: Vacaciones[] = [];

    constructor(private messageService: MessageService,
                private confirmationService: ConfirmationService,
                private authService: AuthService,
                private store: Store<{ rootScope: State }>,
                private utilService: UtilService,
                private vacacionesService: VacacionesService,
                private cd: ChangeDetectorRef,
                private router: Router,
                private fb: FormBuilder) {
        this.dato$ = store.pipe(select('rootScope'));
        this.logged = JSON.parse(localStorage.getItem('logged'));
    }

    filterListaVaca = this.fb.group({
        estado: ''
    })

    FormAddSolicitud = this.fb.group({
        tipo: ['', Validators.compose([Validators.required])],
        cantdis: ['', Validators.compose([Validators.maxLength(2), Validators.max(15), Validators.min(5)])],
        cantdin: ['', Validators.compose([Validators.maxLength(2), Validators.max(15), Validators.min(5)])],
        fecha_ini: ['', Validators.compose([Validators.required])],
        fecha_fin: [''],
        observ: ['', Validators.compose([Validators.required])],
    })

    ngOnInit(): void {
        this.getMyVacations();
    }

    onSort(): void {
    }

    sortData(e: LazyLoadEvent ): void {
        this.sort = e.sortField;
        this.order = e.sortOrder;
        const num = 100;
        let i = 0;
        while (i < num) {
            if (e.rows * i === e.first) {
                this.pageNumber = i + 1;
            }
            i++;
        }
        this.pageSize = e.rows;
        //this.getData();
    }

    /*exportExcel(): void {
        this.utilService.exportExcel(this.empleadosLista, 'Lista empleados');
    }*/

    filterInput(): void {

    }

    updateDates(): void {
        let fecha_ini = this.FormAddSolicitud.controls.fecha_ini.value;
        let cantdis = Number(this.FormAddSolicitud.controls.cantdis.value);
        let cantdin = Number(this.FormAddSolicitud.controls.cantdin.value);
        let cantd = cantdin + cantdis;

        if (cantdis == 0 || cantdis == null) {
            return;
        }

        if (fecha_ini == '' || fecha_ini == null) {
            return;
        }

        if (cantd > 15 ) {
            Swal.fire({
                icon: 'info',
                title: 'Advertencia',
                timer: 1500,
                showConfirmButton: false,
                text: 'Maximo 15 días por solicitud',
            }).then(() => {
                return;
            });
        }

        this.vacacionesService.calcDate(cantdis, fecha_ini).subscribe(data => {
            if (data.state === 200) {
                this.FormAddSolicitud.controls.fecha_fin.setValue(data.msg);
            } else if (data.state === 402) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    if (this.logged) {
                        this.authService.logout().subscribe(datalog => {
                            this.response = datalog;
                            if (this.response.state === 200) {
                                localStorage.removeItem('logged');
                                this.store.dispatch(new fromRoot.CleanRootScope());
                                window.location.href = '/login';
                            } else if (this.response.state === 500) {
                                this.router.navigate(['/error']);
                            }
                        }, error => {
                            this.router.navigate(['/error']);
                        });
                    }
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 401) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 500) {
                this.router.navigate(['/error']);
            } else {
                this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
            }
        }, error => {
            this.router.navigate(['/error']);
        });
    }

    openAddSol(): void {
        this.checkOpenAddSol = true;
    }

    addSolicitud(): void {

        let tipo = this.FormAddSolicitud.controls.tipo.value;

        if (tipo == "T") {
            this.FormAddSolicitud.controls.cantdin.setValue("0");
        } else if (tipo == "D") {
            this.FormAddSolicitud.controls.cantdis.setValue("0");
        }

        let cantdis = Number(this.FormAddSolicitud.controls.cantdis.value);
        let cantdin = Number(this.FormAddSolicitud.controls.cantdin.value);

        if (cantdin > 7) {
            Swal.fire({
                icon: 'info',
                title: 'Advertencia',
                timer: 1500,
                showConfirmButton: false,
                text: 'Maximo 7 días en dinero en una solictud'
            }).then(() => {
                return;
            });
        }

        if (cantdis > 14 && tipo == "M") {
            Swal.fire({
                icon: 'info',
                title: 'Advertencia',
                timer: 1500,
                showConfirmButton: false,
                text: 'Maximo 14 días de disfrute en una solicitud mixta',
            }).then(() => {
                return;
            });
        }

        if (cantdin < 1 && tipo == "M") {
            Swal.fire({
                icon: 'info',
                title: 'Advertencia',
                timer: 1500,
                showConfirmButton: false,
                text: 'Minimo 1 día en dinero en una solicitud mixta',
            }).then(() => {
                return;
            });
        }

        let can = cantdis + cantdin;

        if (can > 15) {
            Swal.fire({
                icon: 'info',
                title: 'Advertencia',
                timer: 1500,
                showConfirmButton: false,
                text: 'Maximo 15 días por solicitud',
            }).then(() => {
                return;
            });
        }

        const dataObj: object = {
            dias_dinero: cantdin,
            dias_disfrute: cantdis,
            tipo: String(tipo),
            fecha_ini: this.FormAddSolicitud.controls.fecha_ini.value,
            fecha_fin: String(this.FormAddSolicitud.controls.fecha_fin.value),
            observ: String(this.FormAddSolicitud.controls.observ.value)
        };

        this.vacacionesService.addNewSolicitud(dataObj).subscribe(data => {
            if (data.state === 200) {
                Swal.fire({
                    icon: 'success',
                    title: 'Exito',
                    timer: 2500,
                    showConfirmButton: false,
                    text: 'Se guardo correctamente la solicitud',
                }).then(() => {
                    this.getMyVacations();
                    this.hideDialog();
                    return;
                });
            } else if (data.state === 402) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    if (this.logged) {
                        this.authService.logout().subscribe(datalog => {
                            this.response = datalog;
                            if (this.response.state === 200) {
                                localStorage.removeItem('logged');
                                this.store.dispatch(new fromRoot.CleanRootScope());
                                window.location.href = '/login';
                            } else if (this.response.state === 500) {
                                this.router.navigate(['/error']);
                            }
                        }, error => {
                            this.router.navigate(['/error']);
                        });
                    }
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 401) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 500) {
                this.router.navigate(['/error']);
            } else {
                this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
            }
        }, error => {
            this.router.navigate(['/error']);
        });
    }

    getMyVacations(): void {
        const dataObj: object = {
            limit: this.pageSize,
            page: this.pageNumber,
            sort: this.sort,
            filter: this.filterListaVaca.value,
            order: this.order
        };
        this.loading = true;
        this.vacacionesService.getMyVacations(dataObj).subscribe(data => {
            this.loading = false;
            if (data.state === 200) {
                this.vacacionesLista = [...data.dataArray];
                this.record = data.dataObjet.records;
                console.log(this.vacacionesLista);
                console.log([...data.dataArray]);
                console.log(this.record);
            } else if (data.state === 402) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    if (this.logged) {
                        this.authService.logout().subscribe(datalog => {
                            this.response = datalog;
                            if (this.response.state === 200) {
                                localStorage.removeItem('logged');
                                this.store.dispatch(new fromRoot.CleanRootScope());
                                window.location.href = '/login';
                            } else if (this.response.state === 500) {
                                this.router.navigate(['/error']);
                            }
                        }, error => {
                            this.router.navigate(['/error']);
                        });
                    }
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 401) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 500) {
                this.router.navigate(['/error']);
            } else {
                this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
            }
        }, error => {
            this.router.navigate(['/error']);
        });
    }


    hideDialog(): any {
        this.checkOpenAddSol = false;
        this.FormAddSolicitud.reset();
    }

}
