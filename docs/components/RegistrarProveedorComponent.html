import {ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {FormBuilder, Validators} from '@angular/forms';
import {ConfirmationService, MessageService} from 'primeng';
import {AuthService} from '@core/services/auth/auth.service';
import {ProveedoresService} from '@core/services/proveerdores/proveedores.service';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {Router} from '@angular/router';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {Observable} from 'rxjs';

@Component({
  selector: 'app-registrar-proveedor',
  templateUrl: './registrar-proveedor.component.html',
  styleUrls: ['./registrar-proveedor.component.scss']
})
export class RegistrarProveedorComponent implements OnInit {

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private proveedoresService: ProveedoresService,
              private store: Store<{ rootScope: State }>,
              private router: Router,
              private cd: ChangeDetectorRef,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };

  dato$: Observable<State>;
  logged: Observable<boolean>;
  submittedMenu = false;
  nombre = '';

  FormAddProveedor = this.fb.group({
    nombre: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    identificacion: ['', Validators.compose([Validators.required, Validators.maxLength(11)])],
    direccion: ['', Validators.compose([Validators.required, Validators.maxLength(30)])],
    ciudad_id: ['', Validators.compose([Validators.required])]
  });

  ngOnInit(): void {
  }

}
