import {AfterViewInit, ChangeDetectorRef, Component, OnInit} from '@angular/core';

interface SubMenu {
  desMenu: string;
  submenu_id?: number;
  descripcion: string;
  pagina: string;
  menu_id: number;
}

  selectMenus(): any {
    this.menuService.getMenuSelect().subscribe(data => {
      this.response = data;
      if (data.state === 200) {
        // this.submenus.push(...this.response.dataArray);
        this.menuSelect = this.response.dataArray;
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  getData(): void {
    const dataObj: object = {
      limit: this.pageSize,
      page: this.pageNumber,
      sort: this.sort,
      filter: this.filterSubMenuForm.value,
      order: this.order
    };
    this.loading = true;
    this.menuService.getSubMenuData(dataObj).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.submenus = [...data.dataArray];
        this.record = data.dataObjet.records;

  updateRowGroupMetaData(): any {
    this.rowGroupMetadata = {};
    if (this.submenus) {
      for (let i = 0; i < this.submenus.length; i++) {
        const rowData = this.submenus[i];
        const brand = rowData.desMenu;
        if (i === 0) {
          this.rowGroupMetadata[brand] = {index: 0, size: 1};
        } else {
          const previousRowData = this.submenus[i - 1];
          const previousRowGroup = previousRowData.desMenu;
          if (brand === previousRowGroup) {
            this.rowGroupMetadata[brand].size++;
          } else {
            this.rowGroupMetadata[brand] = {index: i, size: 1};
          }
        }
      }
    }
  }

  onSort(): void {
    this.updateRowGroupMetaData();
  }

  sortData(e: LazyLoadEvent ): void {
    this.sort = e.sortField;
    this.order = e.sortOrder;
    const num = 100;
    let i = 0;
    while (i < num) {
      if (e.rows * i === e.first) {
        this.pageNumber = i + 1;
      }
      i++;
    }
    this.pageSize = e.rows;
    this.getData();
  }

  exportExcel(): void {
    this.utilService.exportExcel(this.submenus, 'submenu');
  }

  newSubmenu(): any {
    this.subMenuForm.controls.submenu_id.setValue(null);

  }
