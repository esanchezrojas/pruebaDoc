import {ChangeDetectorRef, Component, Input, OnInit} from '@angular/core';
import {ResponseModel} from "@core/models/response-model/ResponseModel";
import {Observable} from "rxjs";
import {State} from "@core/models/store-model/StoreModel";
import {ConfirmationService, LazyLoadEvent, MessageService} from "primeng";
import {AuthService} from "@core/services/auth/auth.service";
import {select, Store} from "@ngrx/store";
import {UtilService} from "@core/services/util/util.service";
import {Router} from "@angular/router";
import {FormBuilder, Validators} from "@angular/forms";
import {VacacionesService} from "@core/services/vacaciones/vacaciones.service";
import Swal from "sweetalert2";
import * as fromRoot from "../../../rootScopeAction";

interface Vacaciones {
    solicitud_id: number;
    empleado_id: number;
    dias_disfrute: number;
    dias_dinero: number;
    jefe_id: number;
    fecha_crea: string;
    tipo: string;
    fecha_ini: string;
    fecha_fin: string;
    fecha_aproba_rechazo: string;
    motivo_coment: string;
    observ: string;
    estado: string;
}

@Component({
    selector: 'app-gestion-jefe',
    templateUrl: './gestion-jefe.component.html',
    styleUrls: ['./gestion-jefe.component.scss']
})
export class GestionJefeComponent implements OnInit {

    @Input() idReq: string;

    response: ResponseModel = {
        dataArray: [],
        msg: '', state: 0
    };

    record = 0;
    display: boolean = false;
    pageSize = 100;
    pageNumber = 1;
    sort: string;
    order: any;
    loading: boolean;
    logged: Observable<boolean>;
    dato$: Observable<State>;
    checkOpenSol: boolean = false;
    submittedMenu: boolean = false;
    checkIdUrl: boolean = false;
    idReqText: string = '';

    vacacionesLista: Vacaciones[] = [];
    solicitud: object;

    constructor(private messageService: MessageService,
                private confirmationService: ConfirmationService,
                private authService: AuthService,
                private store: Store<{ rootScope: State }>,
                private utilService: UtilService,
                private vacacionesService: VacacionesService,
                private cd: ChangeDetectorRef,
                private router: Router,
                private fb: FormBuilder) {
        this.dato$ = store.pipe(select('rootScope'));
        this.logged = JSON.parse(localStorage.getItem('logged'));
    }

    filterListaVaca = this.fb.group({
        estado: '',
        solicitud_id: ''
    })

    FormGesSolicitud = this.fb.group({
        estado: ['', Validators.compose([Validators.required])],
        motivo: ['', Validators.compose([Validators.required])]
    })

    filterInput(): void {
    }

    onSort(): void {
    }

    closeIdUrl(): void {
        this.checkIdUrl = false;
        this.idReq = "0";
        this.filterListaVaca.controls.solicitud_id.setValue('');
    }

    sortData(e: LazyLoadEvent ): void {
        this.sort = e.sortField;
        this.order = e.sortOrder;
        const num = 100;
        let i = 0;
        while (i < num) {
            if (e.rows * i === e.first) {
                this.pageNumber = i + 1;
            }
            i++;
        }
        this.pageSize = e.rows;
        //this.getData();
    }

    ngOnInit(): void {
        this.cargaDataIni();
    }

    cargaDataIni(): void {
        let data = '';
        if (this.idReq === undefined) {
            this.idReq = "0";
        } else {
            this.checkIdUrl = true;
            data = atob(decodeURI(this.idReq));
            data = data.slice(10, -10)
            this.idReqText = data;
            this.filterListaVaca.controls.solicitud_id.setValue(this.idReq);
        }
        this.getSoliGes();
    }

    getSoliGes(): void {
        const dataObj: object = {
            limit: this.pageSize,
            page: this.pageNumber,
            sort: this.sort,
            filter: this.filterListaVaca.value,
            order: this.order
        };
        this.loading = true;
        this.vacacionesService.getSoliGes(dataObj).subscribe(data => {
            this.loading = false;
            if (data.state === 200) {
                this.vacacionesLista = [...data.dataArray];
                this.record = data.dataObjet.records;
                console.log(this.checkOpenSol);
                console.log(this.vacacionesLista[0]);
                if (this.checkIdUrl) {
                    this.openSolModal(this.vacacionesLista[0]);
                }
            } else if (data.state === 402) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    if (this.logged) {
                        this.authService.logout().subscribe(datalog => {
                            this.response = datalog;
                            if (this.response.state === 200) {
                                localStorage.removeItem('logged');
                                this.store.dispatch(new fromRoot.CleanRootScope());
                                window.location.href = '/login';
                            } else if (this.response.state === 500) {
                                this.router.navigate(['/error']);
                            }
                        }, error => {
                            this.router.navigate(['/error']);
                        });
                    }
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 401) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 500) {
                this.router.navigate(['/error']);
            } else {
                this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
            }
        }, error => {
            this.router.navigate(['/error']);
        });
    }

    openSolModal(obj: object): void {

        this.solicitud = obj;
        this.checkOpenSol = true;

    }

    gesSolicitud(sol): void {
        this.submittedMenu = true;
        if (!this.FormGesSolicitud.valid) {
            return;
        }

        let estado = this.FormGesSolicitud.controls.estado.value;
        let motivo = this.FormGesSolicitud.controls.motivo.value;
        let solicitud = sol;

        const dataObj: object = {
            estado: estado,
            motivo: motivo,
            solicitud_id: solicitud
        };

        this.loading = true;
        this.vacacionesService.gestionSol(dataObj).subscribe(data => {
            this.loading = false;
            if (data.state === 200) {
                Swal.fire({
                    icon: 'success',
                    title: 'Exito',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    this.checkOpenSol = false;
                    this.getSoliGes();
                    this.FormGesSolicitud.reset();
                });
            } else if (data.state === 402) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    if (this.logged) {
                        this.authService.logout().subscribe(datalog => {
                            this.response = datalog;
                            if (this.response.state === 200) {
                                localStorage.removeItem('logged');
                                this.store.dispatch(new fromRoot.CleanRootScope());
                                window.location.href = '/login';
                            } else if (this.response.state === 500) {
                                this.router.navigate(['/error']);
                            }
                        }, error => {
                            this.router.navigate(['/error']);
                        });
                    }
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 401) {
                Swal.fire({
                    icon: 'info',
                    title: 'Advertencia',
                    timer: 1500,
                    showConfirmButton: false,
                    text: data.msg,
                }).then(() => {
                    this.router.navigate(['/home']);
                });
            } else if (data.state === 500) {
                this.router.navigate(['/error']);
            } else {
                this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
            }
        }, error => {
            this.router.navigate(['/error']);
        });

    }

    hideDialog(): void {
        this.checkOpenSol = false;
    }

}
