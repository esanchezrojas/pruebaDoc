import {ChangeDetectorRef, Component, OnInit} from '@angular/core';
import { ConfirmationService, MessageService } from 'primeng';
import { ResponseModel } from '@core/models/response-model/ResponseModel';
import { ProveedoresService } from '@core/services/proveerdores/proveedores.service';
import {FormBuilder, Validators} from '@angular/forms';
import { AuthService } from '@core/services/auth/auth.service';
import { select, Store } from '@ngrx/store';
import { State } from '@core/models/store-model/StoreModel';
import { Router } from '@angular/router';
import { Observable } from 'rxjs';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';
import {EmpleadosService} from '@core/services/empleados/empleados.service';
import {Proveedor} from '../../interfaces/proveedor';

export interface TreeNode {
  data?: any;
  children?: TreeNode[];
  leaf?: boolean;
  expanded?: boolean;
}

interface Ciudad {
  ciudad_id: number;
  descripcion: string;
}

@Component({
  selector: 'app-listar-proveedor',
  templateUrl: './listar-proveedor.component.html',
  styleUrls: ['./listar-proveedor.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class ListarProveedorComponent implements OnInit {

  dato$: Observable<State>;
  logged: Observable<boolean>;
  submittedMenu = false;
  submittedContacto = false;
  idProv: number;
  resultsCity: Ciudad[] = [];
  tipoContacto = [];
  ciudadId: any;

  checkOpenAddProduct = false;
  checkOpenAddProv = false;
  checkOpenAddContacto = false;
  nameModalAddContacto = '';

  files: TreeNode[];

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };

  constructor(private proveedorService: ProveedoresService,
              private fb: FormBuilder,
              private cb: ChangeDetectorRef,
              private authService: AuthService,
              private empleadoService: EmpleadosService,
              private store: Store<{ rootScope: State }>,
              private router: Router,
              private messageService: MessageService,
              private confirmationService: ConfirmationService) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  filterNavegacionTree = this.fb.group({
    estado: [''],
    area_id: [''],
    cargo_id: ['']
  });

  FormAddProveedor = this.fb.group({
    nombre: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    identificacion: ['', Validators.compose([Validators.required, Validators.maxLength(11)])],
    direccion: ['', Validators.compose([Validators.required, Validators.maxLength(30)])],
    ciudad_id: ['', Validators.compose([Validators.required])]
  });

  FormAddContacto = this.fb.group({
    proveedor_id: [''],
    tipo_contacto: ['', Validators.compose([Validators.required])],
    nombre: ['', Validators.compose([Validators.required, Validators.maxLength(100)])],
    telefono: ['', Validators.compose([Validators.required, Validators.maxLength(11)])],
    email: ['', Validators.compose([Validators.required, Validators.maxLength(150), Validators.email])]
  });

  ngOnInit(): void {
    this.dataTree();
    this.getClasificadorByType();
  }

  getClasificadorByType(): void {
    this.empleadoService.getClasificadorByType('CONT').subscribe(data => {
      if (data.state === 200) {
        this.tipoContacto = data.dataArray;
        console.log(this.tipoContacto);
      }
    });
  }

  openAddProduct(id: number): void {
    this.checkOpenAddProduct = true;
    this.idProv = id;
  }

  openAddProveedor(): void {
    this.checkOpenAddProv = true;
  }

  openAddContacto(nameProv: string, provId: string): void {
    this.FormAddContacto.controls.proveedor_id.setValue(provId);
    this.nameModalAddContacto = 'Agregar contacto al proveedor ' + nameProv;
    this.checkOpenAddContacto = true;
  }

  closeAllModal(): void {
    this.checkOpenAddProv = false;
    this.checkOpenAddProduct = false;
    this.checkOpenAddContacto = false;
  }

  addNewProveedor(): void {
    this.submittedMenu = true;
    if (!this.FormAddProveedor.valid) {
      return;
    }
    const datos = this.FormAddProveedor.value;
    this.proveedorService.saveProveedor(datos as Proveedor).subscribe(data => {
      this.response = data;
      if (data.state === 200) {
        Swal.fire({
          icon: 'success',
          title: 'Exito',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.dataTree();
          this.closeAllModal();
        });
      }
      else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  addNewContacto(): void {
    this.submittedContacto = true;
    if (!this.FormAddContacto.valid) {
      return;
    }
    const datos = this.FormAddContacto.value;
    this.proveedorService.saveContacto(datos).subscribe(data => {
      this.response = data;
      if (data.state === 200) {
        Swal.fire({
          icon: 'success',
          title: 'Exito',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.dataTree();
          this.closeAllModal();
        });
      }
      else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  dataTree(): void {
    this.proveedorService.getProveedores().then(response => {
      this.files = (response.data as TreeNode[]);
    });
    // this.files = (this.fil.dataObjet.data as TreeNode[]);
    console.log(this.files);
  }

  searchCity(event): void {
    this.empleadoService.getCiudad(event.query).subscribe(data => {
      this.resultsCity = data.dataArray;
    });
  }

}
