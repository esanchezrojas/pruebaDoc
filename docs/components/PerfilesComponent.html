import {ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {PerfilesService} from "@core/services/perfiles/perfiles.service";
import {FormBuilder, Validators} from "@angular/forms";
import {select, Store} from "@ngrx/store";
import {ResponseModel} from "@core/models/response-model/ResponseModel";
import {Observable} from "rxjs";
import {State} from "@core/models/store-model/StoreModel";
import {AuthService} from "@core/services/auth/auth.service";
import {Router} from "@angular/router";
import {ConfirmationService, MessageService} from "primeng";
import Swal from "sweetalert2";
import * as fromRoot from "../../../rootScopeAction";

export interface TreeNode {
  data?: any;
  children?: TreeNode[];
  leaf?: boolean;
  expanded?: boolean;
}

@Component({
  selector: 'app-perfiles',
  templateUrl: './perfiles.component.html',
  styleUrls: ['./perfiles.component.scss'],
  providers: [MessageService, ConfirmationService]
})

export class PerfilesComponent implements OnInit {

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };

  dato$: Observable<State>;
  logged: Observable<boolean>;

  checkAddPerfilApp: boolean = false;
  checkAddPerfilMenu: boolean = false;
  checkEditPerfilApp: boolean = false;
  nameModalPerfilApp: string = '';
  nameModalPerfilMenu: string = '';
  nameModalEditPerfilApp: string = '';
  submittedMenu = false;

  selectMenu = [];
  selectProceso = [];

  constructor(private perfilesServices: PerfilesService,
              private fb: FormBuilder,
              private cb: ChangeDetectorRef,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private router: Router,
              private messageService: MessageService,
              private confirmationService: ConfirmationService) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  FormPerfilApp = this.fb.group({
    aplicacion_id: ['', Validators.compose([Validators.required])],
    descripcion: ['', Validators.compose([Validators.required])]
  })

  FormEditPerfilApp = this.fb.group({
    perfilapp_id: ['', Validators.compose([Validators.required])],
    descripcion: ['', Validators.compose([Validators.required])]
  })

  FormPerfilMenu = this.fb.group({
    menu: ['', Validators.compose([Validators.required])],
    insertar: ['', Validators.compose([Validators.required])],
    actualizar: ['', Validators.compose([Validators.required])],
    eliminar: ['', Validators.compose([Validators.required])],
    perfilapp_id: ['', Validators.compose([Validators.required])]
  })

  filterNavegacionTree = this.fb.group({
    proceso: ['']
  })

  files: TreeNode[];

  dataTree(): void {
    this.perfilesServices.getPerfiles(this.filterNavegacionTree.value).then(data => {
      this.files = <TreeNode[]>data.data;
    });
  }

  getClasificadorByType(app: number): void {
    this.perfilesServices.getSelectMenuApp(app).subscribe(data => {
      if (data.state === 200) {
        this.selectMenu =  data.dataArray;
      }
    });
  }

  getClasificadorByTypePlus(): void {
    this.perfilesServices.getSelectProcesos().subscribe(data => {
      if (data.state === 200) {
        this.selectProceso =  data.dataArray;
      }
    });
  }

  ngOnInit() {
    this.dataTree();
    this.getClasificadorByTypePlus();
  }

  openAddPerfilApp(rowData: any): void {
    this.checkAddPerfilApp = true;
    this.nameModalPerfilApp = 'Crear nuevo perfil para la aplicacin ' + rowData.descripcion;
    this.FormPerfilApp.controls.aplicacion_id.setValue(rowData.aplicacion_id);
  }

  openAddPerfilMenu(rowData: any): void {
    this.checkAddPerfilMenu = true;
    this.nameModalPerfilMenu = 'Agragar permiso al perfil ' + rowData.descripcion + ' en la aplicacion ' + rowData.aplicacion;
    this.getClasificadorByType(rowData.aplicacion_id);
    this.FormPerfilMenu.controls.perfilapp_id.setValue(rowData.perfilapp_id);
  }

  openEditPerfilApp(rowData: any): void {
    this.checkEditPerfilApp = true;
    this.nameModalEditPerfilApp = 'Editar perfil ' + rowData.descripcion + ' en la aplicacin ' + rowData.aplicacion;
    this.FormEditPerfilApp.controls.perfilapp_id.setValue(rowData.perfilapp_id);
    this.FormEditPerfilApp.controls.descripcion.setValue(rowData.descripcion);
  }

  addPerfilApp(): void {
    this.perfilesServices.addDataAppPerfil(this.FormPerfilApp.value).subscribe(data => {
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.FormPerfilApp.controls.descripcion.setValue('');
        this.FormPerfilApp.controls.aplicacion_id.setValue('');
        this.hideDialog();
        this.dataTree();
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      // this.router.navigate(['/error']);
    });
  }

  addPerfilMenu(): void {
    this.submittedMenu = true;
    if(!this.FormPerfilMenu.valid) {
      return;
    }
    this.perfilesServices.addPerfilMenu(this.FormPerfilMenu.value).subscribe(data => {
      if (data.state === 200) {
        this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
        this.submittedMenu = false;
        this.FormPerfilMenu.controls.insertar.setValue('');
        this.FormPerfilMenu.controls.eliminar.setValue('');
        this.FormPerfilMenu.controls.actualizar.setValue('');
        this.FormPerfilMenu.controls.perfilapp_id.setValue('');
        this.FormPerfilMenu.controls.menu.setValue('');
        this.hideDialog();
        this.dataTree();
      } else if (data.state === 402) {
        Swal.fire({

  editEstadoAppPerfil(tipo: string, data: any): void {
    let tipoE: string = 'A';

    if (tipo == 'inactivar') {
      tipoE = 'I';
    }

    this.confirmationService.confirm({
      message: `Estas seguro que quieres ${tipo} el perfil ${data.descripcion} en la aplicacin ${data.aplicacion}`,
      header: 'Cambiar estado perfil',
      icon: 'pi pi-exclamation-triangle',
      acceptLabel: 'Si',
      accept: () => {
        this.perfilesServices.editEstadoPerfilApp(tipoE, data).subscribe(data => {
          if (data.state === 200) {
            this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
            this.dataTree();
          } else if (data.state === 402) {
            Swal.fire({
              icon: 'info',

  editarPermisoPerfilMenu(valor: string, tipo: string, data: any): void {
    let tipoE: string = 'inserte';
    let valorE: string = 'permitir';

    if (tipo == 'A') {
      tipoE = 'actualice';
    } else if ( tipo == 'E' ) {
      tipoE = 'elimine'
    }

    if (valor == 'N') {
      valorE = 'denegar';
    }

    let mensaje: string = `Estas seguro que quieres ${valorE} que el perfil ${data.perfilapp} ${tipoE} informacin en el menu ${data.descripcion}`;

    if (tipo == 'Q') {
      mensaje = `Estas seguro que quieres quitar el permiso para el perfil ${data.perfilapp} en el menu ${data.descripcion}`;
    }

    this.confirmationService.confirm({
      message: mensaje,
      header: 'Cambiar permisos',
      icon: 'pi pi-exclamation-triangle',
      acceptLabel: 'Si',
      accept: () => {
        this.perfilesServices.editPermisoMenu(valor, tipo, data).subscribe(data => {
          if (data.state === 200) {
            this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
            this.dataTree();
          } else if (data.state === 402) {
            Swal.fire({
              icon: 'info',
              title: 'Advertencia',
              timer: 1500,
              showConfirmButton: false,
              text: data.msg,
            }).then(() => {
              if (this.logged) {
                this.authService.logout().subscribe(datalog => {
                  this.response = datalog;
                  if (this.response.state === 200) {
                    localStorage.removeItem('logged');
                    this.store.dispatch(new fromRoot.CleanRootScope());
                    window.location.href = '/login';
                  } else if (this.response.state === 500) {
                    this.router.navigate(['/error']);
                  }
                }, error => {
                  this.router.navigate(['/error']);
                });
              }
              this.router.navigate(['/home']);
            });
          } else if (data.state === 401) {
            Swal.fire({
              icon: 'info',
              title: 'Advertencia',
              timer: 1500,
              showConfirmButton: false,
              text: data.msg,
            }).then(() => {

            });
          } else if (data.state === 500) {

          } else {

          }
        }, error => {
          this.router.navigate(['/error']);
        });
      }
    });
  }

  hideDialog(): void {
    this.checkAddPerfilApp = false;
    this.checkAddPerfilMenu = false;
    this.checkEditPerfilApp = false;
  }

}
