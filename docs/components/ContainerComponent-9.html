import { Component, OnInit } from '@angular/core';
import {Observable} from 'rxjs';
import {State} from '@core/models/store-model/StoreModel';
import {select, Store} from '@ngrx/store';
import {ActivatedRoute, ParamMap, Router} from '@angular/router';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';
import {MenuOptionsService} from '@core/services/menu-options/menu-options.service';

@Component({
  selector: 'app-container',
  templateUrl: './container.component.html',
  styleUrls: ['./container.component.scss']
})
export class ContainerComponent implements OnInit {

  tab: string;
  crear = false;
  listar = false;
  cargarInventario = false;
  dato$: Observable<State>;
  NavMenu: any = [];

  loading = false;

  constructor(private store: Store<{ rootScope: State }>,
              private rutaActiva: ActivatedRoute,
              private router: Router,
              private menuOptionsService: MenuOptionsService) {
    this.dato$ = store.pipe(select('rootScope'));
    this.tab = this.rutaActiva.snapshot.params.sub;
    this.initTapMenu(this.tab);
  }

  ngOnInit(): void {
    this.getNavMenu();
    this.rutaActiva.paramMap.subscribe((params: ParamMap) => {
      this.tab = params.get('sub');
      console.log(params.get('sub'));
      this.getNavMenu();
      this.initTapMenu(this.tab);
    });
  }
  getNavMenu(): void {
    this.menuOptionsService.loadMenuApp(this.router.url).subscribe(data => {
      this.NavMenu = data.dataArray[0];
    }, error => {
      console.log(error);
    });
  }
  initTapMenu(idMenu: string): void {
    switch (idMenu) {
      case 'crear':
        this.crear = true;
        this.listar = false;
        this.cargarInventario = false;
        break;
      case 'listar':
        this.crear = false;
        this.listar = true;
        this.cargarInventario = false;
        break;
      case 'cargar-inventario':
        this.crear = false;
        this.listar = false;
        this.cargarInventario = true;
        break;

    }
  }
  changeTabMenu(e): void {
    localStorage.setItem('tap-menu', JSON.stringify(e.index));
    this.tab = JSON.parse(localStorage.getItem('tap-menu'));
    this.initTapMenu(this.tab);
  }

}
