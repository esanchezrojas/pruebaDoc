import {Component, OnInit} from '@angular/core';
import {Observable} from 'rxjs';
import {State} from '@core/models/store-model/StoreModel';
import {select, Store} from '@ngrx/store';
import {EmpleadosService} from '@core/services/empleados/empleados.service';
import {ActivatedRoute, ParamMap, Router} from '@angular/router';
import {MenuOptionsService} from '@core/services/menu-options/menu-options.service';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';

@Component({
  selector: 'app-container',
  templateUrl: './container.component.html',
  styleUrls: ['./container.component.scss']
})
export class ContainerComponent implements OnInit {

  tab: string;
  nuevo = false;
  nuevoCorto = false;
  misDatos = false;
  reporte = false;
  listar = false;
  procesoA = false;
  actualizacion = false;
  dato$: Observable<State>;
  NavMenu: any = [];

  loading = false;

  constructor(private store: Store<{ rootScope: State }>,
              private empleadoService: EmpleadosService,
              private rutaActiva: ActivatedRoute,
              private router: Router,
              private menuOptionsService: MenuOptionsService) {
    this.dato$ = store.pipe(select('rootScope'));
    this.tab = this.rutaActiva.snapshot.params.sub;
    this.initTapMenu(this.tab);
  }


  ngOnInit(): void {
    this.getNavMenu();
    this.rutaActiva.paramMap.subscribe((params: ParamMap) => {
      this.tab = params.get('sub');
      this.getNavMenu();
      this.initTapMenu(this.tab);
    });
  }

  getNavMenu(): void {
    this.menuOptionsService.loadMenuApp(this.router.url).subscribe(data => {
      this.NavMenu = data.dataArray[0];
      console.log(this.NavMenu);
    }, error => {
      console.log(error);
    });
  }

  initTapMenu(idMenu: string): void {
    switch (idMenu) {
      case 'nuevo':
        this.listar = false;
        this.nuevo = true;
        this.nuevoCorto = false;
        this.procesoA = false;
        this.actualizacion = false;
        this.misDatos = false;
        this.reporte = false;
        break;
      case 'nuevo-corto':
        this.listar = false;
        this.nuevo = false;
        this.nuevoCorto = true;
        this.procesoA = false;
        this.actualizacion = false;
        this.misDatos = false;
        this.reporte = false;
        break;
      case 'listar':
        this.listar = true;
        this.nuevo = false;
        this.nuevoCorto = false;
        this.procesoA = false;
        this.actualizacion = false;
        this.misDatos = false;
        this.reporte = false;
        break;
      case 'proceso-a':
        this.listar = false;
        this.nuevo = false;
        this.nuevoCorto = false;
        this.procesoA = true;
        this.actualizacion = false;
        this.misDatos = false;
        this.reporte = false;
        break;
      case 'actualizacion':
        this.listar = false;
        this.nuevo = false;
        this.nuevoCorto = false;
        this.procesoA = false;
        this.actualizacion = true;
        this.misDatos = false;
        this.reporte = false;
        break;
      case 'mis-datos':
        this.listar = false;
        this.nuevo = false;
        this.nuevoCorto = false;
        this.procesoA = false;
        this.actualizacion = false;
        this.misDatos = true;
        this.reporte = false;
        break;
      case 'reporte':
        this.listar = false;
        this.nuevo = false;
        this.nuevoCorto = false;
        this.procesoA = false;
        this.actualizacion = false;
        this.misDatos = false;
        this.reporte = true;
        break;
    }
  }

  changeTabMenu(e): void {
    localStorage.setItem('tap-menu', JSON.stringify(e.index));
    this.tab = JSON.parse(localStorage.getItem('tap-menu'));
    this.initTapMenu(this.tab);
  }

}
