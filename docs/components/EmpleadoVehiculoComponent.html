import {ChangeDetectorRef, Component, OnInit} from '@angular/core';
import {ConfirmationService, MessageService} from 'primeng';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import {Observable} from 'rxjs';
import {State} from '@core/models/store-model/StoreModel';
import {AuthService} from '@core/services/auth/auth.service';
import {select, Store} from '@ngrx/store';
import {MenuService} from '@core/services/menu/menu.service';
import {UtilService} from '@core/services/util/util.service';
import {EmpleadosService} from '@core/services/empleados/empleados.service';
import {Router} from '@angular/router';
import {FormBuilder, Validators} from '@angular/forms';
import Swal from 'sweetalert2';
import * as fromRoot from '../../../rootScopeAction';

interface Vehiculo {
  empleadovehiculo_id: number;
  empleado_id: number;
  tipovehiculo_id: number;
  marca: string;
  referencia: string;
  modelo: string;
  placa: string;
  estado: string;
}

@Component({
  selector: 'app-empleado-vehiculo',
  templateUrl: './empleado-vehiculo.component.html',
  styleUrls: ['./empleado-vehiculo.component.scss'],
  providers: [MessageService, ConfirmationService]
})
export class EmpleadoVehiculoComponent implements OnInit {

  constructor(private messageService: MessageService,
              private confirmationService: ConfirmationService,
              private authService: AuthService,
              private store: Store<{ rootScope: State }>,
              private menuService: MenuService,
              private utilService: UtilService,
              private empleadoService: EmpleadosService,
              private cd: ChangeDetectorRef,
              private router: Router,
              private fb: FormBuilder) {
    this.dato$ = store.pipe(select('rootScope'));
    this.logged = JSON.parse(localStorage.getItem('logged'));
  }

  response: ResponseModel = {
    dataArray: [],
    msg: '', state: 0
  };
  dato$: Observable<State>;
  logged: Observable<boolean>;
  loading = true;
  submittedForm = false;
  textSpan = 'Agregar vehículo';
  accion = 'ADD';
  pageSize = 100;
  pageNumber = 1;
  record = 0;
  sort: string;
  order: any;
  vehiculosLista = [];
  tipoVehiculo = [];
  carDeletedSel = 0;
  checkDeleted = false;
  headTextDelete = '';

  filterListaVehiculos = this.fb.group({
    tipo: 'Admin'
  });

  EmpleadoVehiculo = this.fb.group({
    tipovehiculo_id: ['', Validators.compose( [Validators.required])],
    marca: ['', Validators.compose([Validators.required, Validators.maxLength(20)])],
    referencia: ['', Validators.compose([Validators.required, Validators.maxLength(30)])],
    modelo: ['', Validators.compose( [Validators.required, Validators.min(1980), Validators.max(2030)])],
    placa: ['', Validators.compose( [Validators.required, Validators.maxLength(6)])],
    empleadovehiculo_id: ['']
  });

  getClasificadorByType(): void {
    this.loading = true;
    this.empleadoService.getClasificadorByType('TVEH').subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        this.tipoVehiculo = data.dataArray;
      }
    });
  }

  ngOnInit(): void {
    this.getData();
    this.getClasificadorByType();
  }

  getData(): void {
    const dataObj: object = {
      limit: this.pageSize,
      page: this.pageNumber,
      sort: this.sort,
      filter: this.filterListaVehiculos.value,
      order: this.order
    };
    this.loading = true;
    this.empleadoService.getVehiculosEmpleado(dataObj).subscribe(data => {
      this.loading = false;
      if (data.state === 200) {
        console.log(this.vehiculosLista);
        this.vehiculosLista = [...data.dataArray];
        this.record = data.dataObjet.records;
        console.log(this.vehiculosLista);
        console.log([...data.dataArray]);
        console.log(this.record);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  saveVehiculo(): void {
    this.submittedForm = true;
    if (!this.EmpleadoVehiculo.valid) {
      return;
    }
    if (this.accion === 'ADD') {
      this.empleadoService.addEmpleadoVehiculo(this.EmpleadoVehiculo.value).subscribe(data => {
        this.saveVehiculoResponse(data);
      }, error => {
        this.router.navigate(['/error']);
      });
    } else if (this.accion === 'EDIT') {
      this.empleadoService.updateEmpleadoVehiculo(this.EmpleadoVehiculo.value).subscribe(data => {
        this.saveVehiculoResponse(data);
      }, error => {
        this.router.navigate(['/error']);
      });
    } else {
      this.router.navigate(['/error']);
    }
  }

  saveVehiculoResponse(data: any): void {
    this.loading = false;
    this.response = data;
    if (data.state === 200) {
      this.messageService.add({severity: 'success', summary: 'Exito', detail: data.msg, life: 3000});
      this.EmpleadoVehiculo.reset();
      this.submittedForm = false;
      this.accion = 'ADD';
      this.textSpan = 'Agregar hijo';
      this.getData();
      this.hideDialog();
    } else if (data.state === 402) {
      Swal.fire({
        icon: 'info',
        title: 'Advertencia',
        timer: 1500,
        showConfirmButton: false,
        text: data.msg,
      }).then(() => {
        if (this.logged) {
          this.authService.logout().subscribe(datalog => {
            this.response = datalog;
            if (this.response.state === 200) {
              localStorage.removeItem('logged');
              this.store.dispatch(new fromRoot.CleanRootScope());
              window.location.href = '/login';
            } else if (this.response.state === 500) {
              this.router.navigate(['/error']);
            }
          }, error => {
            this.router.navigate(['/error']);
          });
        }
        this.router.navigate(['/home']);
      });
    } else if (data.state === 401) {
      Swal.fire({
        icon: 'info',
        title: 'Advertencia',
        timer: 1500,
        showConfirmButton: false,
        text: data.msg,
      }).then(() => {
        this.router.navigate(['/home']);
      });
    } else if (data.state === 500) {
      this.router.navigate(['/error']);
    } else {
      this.messageService.add({severity: 'error', summary: 'Advertencia', detail: data.msg, life: 3000});
    }
  }

  openDataEdit(vehi: Vehiculo): void {
    console.log(vehi);
    this.textSpan = 'Editar vehiculo N° ' + vehi.empleadovehiculo_id;
    this.accion = 'EDIT';

    this.EmpleadoVehiculo.controls.marca.setValue(vehi.marca);
    this.EmpleadoVehiculo.controls.tipovehiculo_id.setValue(vehi.tipovehiculo_id);
    this.EmpleadoVehiculo.controls.referencia.setValue(vehi.referencia);
    this.EmpleadoVehiculo.controls.placa.setValue(vehi.placa);
    this.EmpleadoVehiculo.controls.modelo.setValue(vehi.modelo);
    this.EmpleadoVehiculo.controls.empleadovehiculo_id.setValue(vehi.empleadovehiculo_id);
  }

  openDeleteVehiculo(vehi: Vehiculo): void {
    this.carDeletedSel = vehi.empleadovehiculo_id;
    this.headTextDelete = 'Eliminar el vehiculo con placa ' + vehi.placa;
    this.checkDeleted = true;
  }

  hideDialog(): void {
    this.carDeletedSel = 0;
    this.headTextDelete = '';
    this.checkDeleted = false;
  }

  deleteVehiculo(): void {
    this.empleadoService.deleteEmpleadoVehiculo(this.carDeletedSel).subscribe(data => {
      this.saveVehiculoResponse(data);
    }, error => {
      this.router.navigate(['/error']);
    });
  }

}
