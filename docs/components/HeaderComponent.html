import {ChangeDetectorRef, Component, HostListener, OnInit} from '@angular/core';
import {Observable, Subject} from 'rxjs';
import {AuthService} from '@core/services/auth/auth.service';
import {Router} from '@angular/router';
import * as fromRoot from '../../../rootScopeAction';
import {select, Store} from '@ngrx/store';
import {State} from '@core/models/store-model/StoreModel';
import {ResponseModel} from '@core/models/response-model/ResponseModel';
import Swal from 'sweetalert2';
import {MenuOptionsService} from '@core/services/menu-options/menu-options.service';
import {UtilService} from '@core/services/util/util.service';
import {FormBuilder, Validators} from '@angular/forms';
import {CambioContrasenaService} from '@core/services/cambio-contrasena/cambio-contrasena.service';
import $ from 'jquery';

interface NavMenu {
  sub: string;
  menu: string;
  submenu_id: number;
  pagina: string;
  icono: string;

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss']
})
export class HeaderComponent implements OnInit {

  constructor(private authService: AuthService,
              private menuOptionsService: MenuOptionsService,
              private utilsService: UtilService,
              private store: Store<{ rootScope: State }>,
              private router: Router,
              private fb: FormBuilder,
              private cd: ChangeDetectorRef,
              private cambioContrasenaService: CambioContrasenaService) {
    this.dato$ = store.pipe(select('rootScope'));
    this.inactivityTime();
    this.inactivitySession.subscribe(() => this.closeSessionTime());
    this.logged = JSON.parse(localStorage.getItem('logged'));
    const theme = localStorage.getItem('Theme');
    if (theme === 'DARK') {
      $('body').removeClass('themeLight');
    } else {
      $('body').addClass('themeLight');
    }
  }

  response: ResponseModel;
  result: any[];
  navMenu: any;
  activitySession;
  inactivitySession: Subject<any> = new Subject();
  display: boolean;
  public logged: Observable<boolean>;
  view = false;
  dato$: Observable<State>;
  menu = [];
  notiArray = [];
  changePass = false;
  closeModalPass = true;
  pass = '';

  cambioContrasena = this.fb.group({
    contrasena_actual: ['', Validators.compose([
        Validators.required
      ]
    )],
    n_contrasena: ['', Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#\$%\^&\*{}\.-])(?=.{10,50})'))
      ]
    )], c_contrasena: ['', Validators.compose([
        Validators.required,
        Validators.pattern(new RegExp('^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#\$%\^&\*{}\.-])(?=.{10,50})'))
      ]
    )],
    metodo: ['guardarConfiguracion']
  });

  @HostListener('window:mousemove') refreshUserStateMouse(): void {
    clearTimeout(this.activitySession);
    this.inactivityTime();
  }

  @HostListener('document:keypress', ['$event']) refreshUserStateKey(): void {
    clearTimeout(this.activitySession);
    this.inactivityTime();
  }

  inactivityTime(): void {
    this.activitySession = setTimeout(() => this.inactivitySession.next(undefined), 2200000);
  }

  closeSessionTime(): void {
    if (this.logged) {
      this.authService.logout().subscribe(data => {
        this.response = data;
        if (this.response.state === 200) {
          localStorage.removeItem('logged');
          localStorage.removeItem('tok');
          this.store.dispatch(new fromRoot.CleanRootScope());
          Swal.fire({
            icon: 'info',
            title: 'Advertencias',
            timer: 1500,
            showConfirmButton: false,
            text: 'Ha caducado el tiempo de sesión',
          }).then(() => {
            window.location.href = '/login';
            // this.router.navigate(['/login']);
          });
        } else if (this.response.state === 500) {
          this.router.navigate(['/error']);
        }
      }, error => {
        this.router.navigate(['/error']);
      });
    }
  }

  closeSession(): void {
    if (this.logged) {
      this.authService.logout().subscribe(data => {
        this.response = data;
        if (this.response.state === 200) {
          localStorage.removeItem('logged');
          localStorage.removeItem('tok');
          this.store.dispatch(new fromRoot.CleanRootScope());
          Swal.fire({
            icon: 'info',
            title: 'Advertencia',
            timer: 1500,
            showConfirmButton: false,
            text: 'Ha cerrado sesión',
          }).then(() => {
            window.location.href = '/login';
            // this.router.navigate(['/login']);
          });
        } else if (this.response.state === 500) {
          this.router.navigate(['/error']);
        }
      }, error => {
        this.router.navigate(['/error']);
      });
    }
  }

  getNavMenu(): void {
    this.menuOptionsService.getNavMenu().subscribe(data => {
      if (data.state === 200) {
        this.navMenu = data.dataObjet.data;
        console.log(this.navMenu);
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';
              } else if (this.response.state === 500) {
                this.router.navigate(['/error']);
              }
            }, error => {
              this.router.navigate(['/error']);
            });
          }
          this.router.navigate(['/home']);
        });
      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);
        });
      } else if (data.state === 500) {
        this.router.navigate(['/error']);
      } else {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        });
      }
    }, error => {
      console.log(error);
    });
  }

  getNoti(): void {
    this.menuOptionsService.getNotiData().subscribe(data => {
      if (data.state === 200) {
        console.log(data);
        this.notiArray = data.dataArray;
      } else if (data.state === 402) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          if (this.logged) {
            this.authService.logout().subscribe(datalog => {
              this.response = datalog;
              if (this.response.state === 200) {
                localStorage.removeItem('logged');
                this.store.dispatch(new fromRoot.CleanRootScope());
                window.location.href = '/login';

                this.router.navigate(['/error']);

            }, error => {
              this.router.navigate(['/error']);
            });
          }

      } else if (data.state === 401) {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        }).then(() => {
          this.router.navigate(['/home']);

      } else if (data.state === 500) {

        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 1500,
          showConfirmButton: false,
          text: data.msg,
        });
      }
    }, error => {
      console.log(error);
    });
  }

  checkPass(): void {
    this.authService.checkPass().subscribe(data => {
      this.response = data;
      if (this.response.state === 200) {
        if (data.dataObjet.pass === 1) {
          this.changePass = true;
          this.closeModalPass = false;
          return;
        }
        if (data.dataObjet.updated === 1) {
          Swal.fire({
            icon: 'info',
            title: 'Información',
            text: 'Debes llenar tus datos en el módulo "Mis datos", serás redireccionado. Esta actualización es obligatoria.',
            showConfirmButton: true,
          }).then(() => {
            this.router.navigate(['/empleados/mis-datos']);
          });
        }
        if (data.dataObjet.updated === 4) {
          Swal.fire({
            icon: 'info',
            title: 'Información',
            text: 'Ya llenaste tus datos, estos están en proceso de verificación por parte de RRHH, espera la verificación en tu correo electrónico. Mientras esperas, puedes llenar la información de tus hijos, estudios, mascotas y vehículos, en el módulo de Mis datos.',
            showConfirmButton: true,
          }).then(() => {
            this.router.navigate(['/empleados/mis-datos']);
          });
        }
        if (data.dataObjet.updated === 2) {
          Swal.fire({
            icon: 'info',
            title: 'Información',
            text: 'Tu actualización de datos fue rechazada, verífica tus datos en el módulo "Mis datos", serás redireccionado. Esta actualización, es obligatoria.',
            showConfirmButton: true,
          }).then(() => {
            this.router.navigate(['/empleados/mis-datos']);
          });
        }
      } else if (this.response.state === 500) {
        this.router.navigate(['/error']);
      }
    }, error => {
      this.router.navigate(['/error']);
    });
  }

  cambioContrasenaSubmit(): void {
    this.cambioContrasenaService.cambioContrasena(this.cambioContrasena.value).subscribe(data => {
      this.response = data;
      if (data.state === 200) {
        Swal.fire({
          icon: 'success',
          title: '',
          timer: 5000,
          showConfirmButton: false,
          text: data.msg,
        });
        this.onRefresh();
      } else {
        Swal.fire({
          icon: 'info',
          title: 'Advertencia',
          timer: 4000,
          showConfirmButton: false,
          text: data.msg,
        });
      }
    });
  }

  onRefresh(): void {
    this.router.routeReuseStrategy.shouldReuseRoute = () => false;

    const currentUrl = this.router.url + '?';

    this.router.navigateByUrl(currentUrl)
      .then(() => {
        this.router.navigated = false;
        this.router.navigate([this.router.url]);
      });
  }

  switchTheme(): void {
    const getTheme = localStorage.getItem('Theme');
    if (getTheme === 'LIGHT') {
      localStorage.setItem('Theme', 'DARK');
      $('body').removeClass('themeLight');
    } else {
      localStorage.setItem('Theme', 'LIGHT');
      $('body').addClass('themeLight');
    }
  }

  ngOnInit(): void {
    this.getNavMenu();
    this.getNoti();
    this.checkPass();
  }
}
